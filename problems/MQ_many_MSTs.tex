\begin{problem}
  Give an example of a weighted simple graph with exactly 5 different
  minimum weight spanning trees.

\hint There is an example with 5 vertices.

\begin{staffnotes}
Extra credit for verifying there is no example with fewer than 5 vertices.
\end{staffnotes}

\examspace[1.5in]

\begin{solution}
The length 5 cycle, $C_5$ with all edges of equal weight.
\end{solution}
\end{problem}

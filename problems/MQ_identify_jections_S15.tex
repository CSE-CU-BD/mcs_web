\begin{problem}
 
For the following functions on the natural numbers $\mathbb{N}$, indicate whether they are: 
a bijection (\textbf{B}), a surjection but not a bijection (\textbf{S}), an injection but not a bijection (\textbf{I}), or neither an injection nor a surjection (\textbf{N}), and why. 

\begin{enumerate}[i.]
\item $f(x) = x^4$ \instatements{\hfill\examrule[0.4i\n]} 
\examspace[2in]
\item $g(x) = 2x$ \instatements{\hfill\examrule[0.4i\n]}
\examspace[2in]
\item $h(x) = -x$ \instatements{\hfill\examrule[0.4i\n]}
\examspace[2in]
\item $j(x) = \lfloor \frac{x}{2} \rfloor$ \instatements{\hfill\examrule[0.4i\n]}
\examspace[2in]
\end{enumerate}

Give an example of a domain and codomain pairing that would make $j(x)$ a bijection. \\
\textbf{Domain:} \examspace[1in]
\textbf{Codomain:} \examspace[1in]
\end{problem}

\begin{solution}
\begin{enumerate}[i.]
\item \textbf{N}. $(-1)^4 = 1^4$, so $f$ is not an injection. $f$ is not a surjection because there is no $x$ such that $f(x) = -1$. 
\item \textbf{I}. $g$ is not surjective because there is no $x$ that makes $g$ take on odd values. 

\item \textbf{B}. Every integer has one and only one negative.

\item \textbf{S}. $h$ is not injective because $h(4) = h(5) = 2$.

\end{enumerate}

One possible solution is the domain of even integers and the codomain of  all integers. 

\end{solution}

\begin{problem}

%LOOKING TO ADD HORIZONTAL LINES FOR STUDENT ANSWERS -emshen 2/18/15 3AM 
For the following functions on $\mathbb{N}$ (natural numbers), indicate whether they are: 
a bijection (\textbf{B}), a surjection but not a bijection (\textbf{S}), an injection but not a bijection (\textbf{I}), or neither an injection nor a surjection (\textbf{N}), and why. 

\bparts

\ppart $f(x) = x^4$

\begin{solution}
\textbf{N}. $(-1)^4 = 1^4$, so $f$ is not an injection. $f$ is not a surjection because there is no $x$ such that $f(x) = -1$.
\end{solution}

\examspace[2in]

\ppart  $g(x) = 2x$

\begin{solution}
\textbf{I}. $g$ is not surjective because there is no $x$ that makes $g$ take on odd values. 
\end{solution}

\examspace[2in]

\ppart $h(x) = -x$

\begin{solution}
\textbf{B}. Every integer has one and only one negative.
\end{solution}

\examspace[2in]

\ppart $j(x) = \lfloor \frac{x}{2} \rfloor$
\examspace[2in]
\begin{solution}
\textbf{S}. $h$ is not injective because $h(4) = h(5) = 2$.
\end{solution}

\examspace[2in]

\ppart 
Give an example of a domain and codomain pairing that would make $j(x)$ a bijection. \\
\textbf{Domain:} \examspace[1in]\\
\textbf{Codomain:} \examspace[1in]

\begin{solution}
One possible solution is the domain of even integers and the codomain of  all integers. 
\end{solution}

\begin{problem}
The \term{expected absolute deviation} of a real-valued random
variable, $R$, with mean $\mu$, is defined to be
\[
\expect{\, \abs{R - \mu}\, }.
\]
Prove that the expected absolute deviation is always less than or
equal to the standard deviation, $\sigma$.  (For simplicity, you may
assume that $R$ is defined on a finite sample space.)

\hint Suppose the sample space outcomes are $\omega_1, \omega_2,
\dots, \omega_n$, and let
\begin{align*}
\mathbf{p} & \eqdef
  (\sqrt{\prob{\omega_1}}, \sqrt{\prob{\omega_2}}, \dots, \sqrt{\prob{\omega_n}}),\\
\mathbf{r} & \eqdef
(\, \abs{R(\omega_1)-\mu}\sqrt{\prob{\omega_1}},
    \abs{R(\omega_2)-\mu}\sqrt{\prob{\omega_2}}, \dots,
    \abs{R(\omega_n)-\mu}\sqrt{\prob{\omega_n}}\, ).
\end{align*}
As usual, let $\mathbf{v} \cdot \mathbf{w}$ denote the \idx{dot product} of
the $n$-vectors $\mathbf{v}$ and $\mathbf{w}$, and let
$\abs{\mathbf{v}}$ be the \idx{norm} of $\mathbf{v}$, namely,
$\sqrt{\mathbf{v} \cdot \mathbf{v}}$.

Then verify that
\begin{align*}
\lefteqn{\text{expected absolute deviation of $R$}}\\
  & = \mathbf{r} \cdot \mathbf{p}\\
\abs{\mathbf{r}} & = \sigma\\
\abs{\mathbf{p}} & = 1.
\end{align*}

\begin{solution}
\begin{align}
\mathbf{r} \cdot \mathbf{p}
    & \eqdef \sum_{i=1}^n
        [\abs{R(\omega_i)-\mu}\sqrt{\prob{\omega_i}}]\sqrt{\prob{\omega_i}}\notag\\
    & = \sum_{i=1}^n \abs{R(\omega_i)-\mu}\prob{\omega_i}\notag\\
    & = \expect{\, \abs{R - \mu}\, }.\label{=rdotp}
\end{align}

\begin{align}
\abs{\mathbf{r}}
    & \eqdef \sqrt{\sum_{i=1}^n
        \brac{\abs{R(\omega_i)-\mu}\sqrt{\prob{\omega_i}}}^2}\notag\\
    & = \sqrt{\sum_{i=1}^n
        \abs{R(\omega_i)-\mu}^2\prob{\omega_i}}\notag\\
    & = \sqrt{\sum_{i=1}^n
        (R(\omega_i)-\mu)^2\prob{\omega_i}}\notag\\
    & = \sqrt{\expect{R(\omega_i)-\mu}^2}\notag\\
    & = \sqrt{\variance{R}} = \sigma.\label{=absrsig}
\end{align}



\begin{align}
\abs{\mathbf{p}}
   & \eqdef \sqrt{\sum_{i=1}^n \paren{\sqrt{\prob{\omega_i}}}^2}\notag\\
   & = \sqrt{\sum_{i=1}^n \prob{\omega_i}}\notag\\
   & = \sqrt{1} = 1.\label{=absp1}
\end{align}

Now, we have
\begin{align}
\lefteqn{\text{expected absolute deviation of $R$}}
  & = \mathbf{r} \cdot \mathbf{p}
     & \text{(by~\eqref{=rdotp})}\\
  & \leq \abs{\mathbf{r}} \abs{\mathbf{p}}
     & \text{(Cauchy-Schwarz inequality)}\\
  & = \sigma \cdot 1
     & \text{(by~\eqref{=absrsig} and~\eqref{=absp1})}\\
  & = \sigma.
\end{align}


\end{solution}
\end{problem}

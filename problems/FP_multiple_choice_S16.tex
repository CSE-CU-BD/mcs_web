\ppart\label{gcdtf} The following statements about the \textbf{greatest common divisor}:

\begin{itemize}

\item If $\gcd(a, b) \neq 1$ and $\gcd(b, c) \neq 1$, then $\gcd(a, c) \neq 1$. \instatements{\hfill 
\True \qquad  \False} \examspace[0.4in]

\begin{solution}
\False, $a=2\cdot 3, b=3\cdot 5, c=5\cdot 7$
\end{solution}

\item If $a \divides b c$ and $\gcd(a, b) = 1$, then $a \divides c$.  \instatements{\hfill 
\True \qquad \False} \examspace[0.4in]

\begin{solution}
\True.
\end{solution}

\item $\gcd(a^n,b^n) =  (\gcd(a,b))^n$  \instatements{\hfill 
\True \qquad \False} \examspace[0.4in]

\begin{solution}
\True.
\end{solution}

\item $\gcd(ab, ac) = a \gcd(b, c)$.  \instatements{\hfill 
\True \qquad  \False} \examspace[0.4in]

\begin{solution}
\True.
\end{solution}

\item $\gcd(1 + a, 1 + b) = 1 + \gcd(a, b)$.  \instatements{\hfill 
\True \qquad  \False} \examspace[0.4in]

\begin{solution}
\False, $a=1,b=2$.
\end{solution}

\item If an integer linear combination of $a$ and $b$ equals 1, then
  so does some integer linear combination of $a$ and $b^2$. \instatements{\hfill
  \True \qquad \False} \examspace[0.4in]

\begin{solution}
\True.
\end{solution}

\item If an integer linear combination of $a$ and $b$ equals 2, then
 so does some integer linear combination of $a$ and
  $b^2$. \instatements{\hfill \True \qquad \False} \examspace[0.4in]

\begin{solution}
\False, let $a=4, b=2$.
\end{solution}

\end{itemize}

\ppart The following statements about \textbf{trees}:

\begin{itemize}

\item Any connected subgraph is a tree.  \instatements{\hfill \True \qquad
  \False} \examspace[0.4in]

\begin{solution}
\True
\end{solution}

\item Adding an edge between two nonadjacent vertices creates a
  cycle. \instatements{\hfill \True \qquad \False} \examspace[0.4in]

\begin{solution}
\True
\end{solution}

%% \item The number of vertices is one less than twice the number of
%%   leaves.  \instatements{\hfill \True \qquad
%%   \False} \examspace[0.4in]

\begin{solution}
  \False.  This property holds for full binary trees, but not
  in general.  A tree with two vertices is a counterexample.
\end{solution}

\item The number of leaves in a tree is not equal to the number of
  non-leaf vertices.  \instatements{\hfill \True \qquad
  \False} \examspace[0.4in]

\begin{solution}
  \False.  A line graph with 4 vertices has 2 non-leaf
  vertices and 2 leaves.
\end{solution}

\item Any subgraph of a tree is a tree.  \instatements{\hfill \True \qquad
  \False} \examspace[0.4in]

\begin{solution}
\textbf{false.  Choose a subgraph with 2 vertices and no edges.}
\end{solution}

%S11 %cut from Monday version
\item The number of vertices in a tree is one less than the number of
  edges.  \instatements{\hfill \True \qquad \False} \examspace[0.4in]

\begin{solution}
  \False.  This got ``edges'' and ``vertices'' reversed.
Every tree is a counterexample.
\end{solution}

\end{itemize}

\ppart\label{simgrpart} The following statements about finite \textbf{simple graphs}:

\begin{itemize}
\item  It has a spanning tree.  \instatements{\hfill \True \qquad
  \False} \examspace[0.4in]

\begin{solution}
\textbf{false.  Any disconnected graph is a counterexample.}
\end{solution}

\item $\card{\vertices{G}} = O(\card{\edges{G}})$ for connected $G$.  \instatements{\hfill \True \qquad
  \False} \examspace[0.4in]

\begin{solution}
\textbf{true.}
\end{solution}

\item $\card{\edges{G}} = O(\card{\vertices{G}})$.  \instatements{\hfill \True \qquad
  \False} \examspace[0.4in]

\begin{solution}
\textbf{false.}  $\card{\vertices{K_n}} = n = o(n^2)$, but
  $\card{\edges{K_n}} = \Theta(n^2)$.

\end{solution}

\item The chromatic number $\chi(G) \leq \max\set{\degr{v} \suchthat
  v \in \vertices{G}}$.  \instatements{\hfill \True \qquad
  \False} \examspace[0.4in]

\begin{solution}
\textbf{false.}  $\chi(K_n) = n > n-1 = \text{ max degree of } v \in \vertices{K_n}$
\end{solution}

\item The chromatic number $\card{\edges{G}} = O(\chi(G))$.  \instatements{\hfill \True \qquad
  \False} \examspace[0.4in]

\begin{solution}
\textbf{false.}   $\chi(K_n) = n$, but $\card{\edges{K_n}} = \Theta(n^2)$.
\end{solution}

\end{itemize}

\ppart A \term{sink} in a random walk digraph is a vertex with no
edges leaving it.  Circle all of the following assertions that are
true of \textbf{stable distributions on random walk graphs} with
exactly two sinks:

\begin{itemize}

\item there may not be any

\item there may be a unique one

%\item there are exacty two

%\item there may be a countably infinite number

\item there may be an uncountable number

\item there always is an uncountable number

\end{itemize}

\begin{solution}
The first two choices are false, and the last two are true.
That's because a distribution in which one sink has probability $r \in
[0,1] \subseteq \reals$ and the other sink has probability $1-r$ is
stable, and there are an is uncountable number of real numbers in $[0,1]$.
\end{solution}

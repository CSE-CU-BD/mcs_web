\begin{problem}
The notation $[x^n]F(x)$ refers to the coefficient of $x^n$ in the
generating function $F(x)$.  Circle all the exprssions below that
equal $[x^n] 4xG(x)$ (most of them do).
\[\begin{array}{lll}
  4[x^n] xG(x)  \qquad &  4x[x^{n}]G(x) \qquad & [x^n]2xG(x) + 2[x^n]xG(x)\\
\\
  ([x^n]4x) \cdot [x^n]G(x) \qquad &  [x]4x \cdot [x^n]xG(x) \qquad & [x^{n+1}]4x^2G(x)
\end{array}\]

\iffalse

\[\begin{array}{|c|c|c|}
\hline
  4[x^n] xG(x)  &  4x[x^{n}]G(x) & [x^n]2xG(x) + 2[x^n]xG(x)\\
\\
\hline
  ([x^n]4x) \cdot [x^n]G(x) &  [x]4x \cdot [x^n]xG(x) & [x^{n+1}]4x^2G(x)\\
\\
\hline
\end{array}\]
\fi

\begin{solution}
  [x^{n-1}] 4G(x), \quad 4[x^{n-1}] G(x), \quad [x^n]2xG(x) +
  2[x^n]xG(x), \quad [x]4x, \cdot [x^n]xG(x) \quad [x^{n+1}]x^2G(x).
\end{solution}
\end{problem}

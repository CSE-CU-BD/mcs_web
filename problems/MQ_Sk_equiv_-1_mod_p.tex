\begin{problem}
  Let $S_k = 1^k + 2^k + \ldots + p^k$, where $p$ is an odd prime and
  $k$ is a positive multiple of $p - 1$.  Find $a$ such that 
  $S_k \equiv a \pmod{p}$ and $-p<a\leq 0$.

\begin{solution}
By Fermat's Theorem, $x^{p-1} \equiv 1 \pmod{p}$
when $1 \leq x \leq p - 1$.  Since $k$ is a multiple of $p-1$,
raising each side to a suitable power proves that $x^k \equiv 1
\pmod{p}$.  Also, since $k$ is positive, $p^k \equiv 0 \pmod{p}$.

Thus:
%
\begin{align*}
S_k & \equiv 1^k + 2^k + \ldots + (p-1)^k + p^k \pmod{p} \\
    & \equiv \underbrace{1 + 1 + \ldots + 1}_{\text{$p-1$ terms}} 
+ 0 \pmod{p} \\
    & \equiv p - 1\pmod{p} \\
    & \equiv - 1 \pmod{p}
\end{align*}
Therefore $a=-1$.
\end{solution}

\end{problem}

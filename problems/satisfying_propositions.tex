\begin{problem}

For the following propositions, indicate whether they are valid (V), satisfiable but not valid (S), or unsatisfiable (N). For propositions you mark S, give a) an assignment of truth values that would satisfy the proposition and b) an assignment of truth values that would not satisfy the proposition.
 
 \begin{enumerate}
 \item{$P \Rightarrow \overline{P}$}
 \item{$P$ AND $Q$  AND $Z$ AND $(P \Rightarrow \overline{Q})$}
  \item{($P$ AND $Q$ AND $(\overline{P}$ OR $\overline{Q})) \Rightarrow (Z$ AND $(P \Rightarrow Q)$ OR $P $ AND $ \overline{P})$}
 \item{$P$ AND $Q$ OR $(P \Rightarrow Q)$}
 \item{$P$ OR $(\overline{P}$ OR $Q) \Rightarrow (Q$ AND $Q \Rightarrow \overline{Q})$}
 \end{enumerate}

\begin{solution}
 \begin{enumerate}
\item{S. a) $P$ is False b) $P$ is True)}
\item{N, since $P$ AND $P \Rightarrow \overline{Q}$ means $Q$ must be False, but we also know it must be True.}
\item{V, since the left-hand side is always False}
\item{S. a) Both are True, or $P$  is false and $Q$ is anything b) $Q$ is False, $P$ is True}
\item{N, since the left-hand side is always True (because of $P$ OR $\overline{P}$) but the right-hand side must be False.}
 \end{enumerate}
\end{solution}

\end{problem}

\begin{problem}
Suppose $p$ is a prime and $p \geq 5$.  This problem will show that
$p^2-1$ is divisible by $24$.

\bparts

\ppart\label{ppm13} Prove that $p \equiv \pm 1 \pmod 3$.

\begin{solution}
The prime $p$ is cannot be divisible by 2 or 3, so $p \equiv 1\text{
  or }2 \pmod 3$.  But $2 \equiv -1 \pmod 3$.
\end{solution}

\ppart Conclude that $p^2-1$ is divisible by 3.

\begin{solution}
It follows immediately from part~\eqref{ppm13} that
\[
p^2-1 \equiv (\pm 1)^2 -1 = 0 \pmod 3,
\]
which means that $p^2-1$ is multiple of 3.
\end{solution}
\eparts

Since $p$ is not divisible by 2, it can be expressed as $2n+1$
for some integer $n >1$.

\bparts

\ppart Show that if $n$ is even, then $p^2-1$ is divisible by 8.

\begin{solution}
\begin{equation}\label{p214n2}
p^2-1 = 4n^2+4n.
\end{equation}
Since $n$ is even, the right hand side of~\eqref{p214n2} is
obviously divisible by $8$.
\end{solution}

\ppart Show that also if $n$ is odd, then $p^2-1$ is divisible by 8.

\begin{solution}
Now $n=2m+1$ for some $m \in \integer^+$, so the right side
of~\eqref{p214n2} equals
\[
4(2m+1)^2 +4(2m+1) = 8m^2+24m+8
\]
which is divisible by $8$.
\end{solution}

\ppart Conclude that $p^2-1$ is divisible by $24$.
\begin{solution}
The previous parts imply that $p^2-1$ is divisible by $8$ and $3$, and
since $8$ and $3$ are relatively prime, it is divisible by their
product $24$.
\end{solution}

\eparts

\end{problem}

\begin{problem}
What is $\remainder(3^{101},10)$? \examrule
\inhandout{(No explanation is required, but no part credit without an
  explanation.)}

\begin{solution}
\textbf{18}.

\begin{align*}
\remainder(3^2,21) & = 9\\
\remainder(3^3,21) & = 6\\
\remainder(3^4,21) & = 18\\
\remainder(3^5,21) & = 12\\
\remainder(3^6,21) & = 15\\
\remainder(3^7,21) & = 3.
\end{align*}

So
\begin{align*}
\remainder(3^{101},21)
  & = \remainder(3^{3 + 7\cdot 14},21)\\
  & = \remainder(3^3 \cdot 3^{7\cdot 14},21)\\
  & = \remainder(3^3 \cdot 3,21)\\
  & = 18.
\end{align*}
\end{solution}
\end{problem}

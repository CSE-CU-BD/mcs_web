\begin{problem}\label{generprob}
   Prove that $\sqrt[4]{27}$ is irrational.

\begin{solution}
\begin{proof}
  Assume for the sake of contradiction that $\sqrt[4]{27}$ is rational.
  Under this assumption, there exist integers $a$ and $b$ such that
\[
\sqrt[4]{27} =  a/b,
\]
where $a$ and $b$ have no common factor.  Now we prove that $a$ and $b$
are both even, that is, they have 2 as a common factor, a contradiction.

\begin{align*}
\sqrt[4]{27} & = \frac{a}{b}, & \text{(by assumption)}\\
27           & = \frac{a^{4}}{b^{4}}, & \text{(cubing both sides)}\\
27b^{4}      & = a^{4}.
\end{align*}
The lefthand side of the last equation is even, so $a^{4}$ is even, which
implies that $a$ is even.

In particular, $a = 2c$ for some integer $c$.  Thus,
\begin{align*}
27b^{4} & = (2c)^4 = 8c^{4},\\
b^{4}  & = 2c^{4} & \text{(dividing by 27)}.
\end{align*}
The righthand side of the last equation is even,
so $b^{4}$ is even, which implies that $b$  is even.
\end{proof}

\end{solution}

\end{problem}

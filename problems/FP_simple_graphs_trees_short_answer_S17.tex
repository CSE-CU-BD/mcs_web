\begin{problem} 
Let $C_{41}$ be the graph with vertices $\set{0,1,\dots,40}$ and edges
$\edge{0}{1}, \edge{1}{2}, \dots, \edge{39}{40}, \edge{40}{0}$, and
  let $K_{41}$ be the \emph{complete graph} on the same set of 41
  vertices.

You may answer the following questions with formulas involving
exponents, binomial coefficients, and factorials.

\bparts

\ppart How many edges are there in $K_{41}$?

\begin{center}
\exambox{0.7in}{0.5in}{-0.1in}
\end{center}

\begin{solution}
\[
\binom{41}{2} = 820.
\]
\end{solution}

\ppart How many isomorphisms are there from $K_{41}$ to $K_{41}$?

\begin{center}
\exambox{0.7in}{0.5in}{-0.1in}
\end{center}

\begin{solution}
$41!$.

Every bijection from $\vertices{K_{41}}$ to $\vertices{K_{41}}$ is an
isomorphism.
\end{solution}

\ppart How many isomorphisms are there from $C_{41}$ to $C_{41}$?

\begin{center}
\exambox{0.7in}{0.5in}{-0.1in}
\end{center}

\begin{solution}
$2 \cdot 41 = 82$.

Every rotation of $C_{41}$, and every rotation of the reflection of
$C_{41}$ about a fixed axis, is an isomorphism.

For example, the clockwise length-$k$ rotation of $C_{41}$ is the
function
\[
f_k(n) \eqdef n + k \inzmod{41},
\]
and $-f_k$ is the length-$k$ rotation of the reflection of $C_{41}$
for $k \in \Zintvco{0}{41}$.  Moreover, these are the only
isomorphisms.
\end{solution}

\ppart What is the chromatic number $\chi(K_{41})$?

\begin{center}
\exambox{0.7in}{0.5in}{-0.1in}
\end{center}

\begin{solution}
41.

All vertices are adjacent, so they must have different colors.
\end{solution}

\ppart What is the chromatic number $\chi(C_{41})$?

\begin{center}
\exambox{0.7in}{0.5in}{-0.1in}
\end{center}

\begin{solution}
\textbf{3}.

As observed in Section~\bref{sec:coloring},
\begin{align*}
\chi(C_{\text{even}}) & = 2,\\
\chi(C_{\text{odd}})  & = 3.
\end{align*}
\end{solution}

\ppart How many edges are there in a spanning tree of $K_{41}$?

\begin{center}
\exambox{0.7in}{0.5in}{-0.1in}
\end{center}

\begin{solution}
\textbf{40}.

A tree with 41 vertices has 40 edges.
\end{solution}

\inbook{\ppart A graph is created by adding a single edge between
  nonadjacent vertices of a tree with 41 vertices.  What is the
  largest number of cycles the graph might have?

\begin{center}
\exambox{0.7in}{0.5in}{-0.1in}
\end{center}

\begin{solution}
Exactly \textbf{1}.
\end{solution}

\ppart What is the smallest number of leaves possible in a spanning
tree of $K_{41}$?

\begin{center}
\exambox{0.7in}{0.5in}{-0.1in}
\end{center}

\begin{solution}
\textbf{2}.
\end{solution}

\ppart What is the largest number of leaves possible in a in a
  spanning tree of $K_{41}$?

\begin{center}
\exambox{0.7in}{0.5in}{-0.1in}
\end{center}

\begin{solution}
\textbf{40}.
\end{solution}
}

\ppart How many spanning trees does $C_{41}$ have?

\begin{center}
\exambox{0.7in}{0.5in}{-0.1in}
\end{center}

\begin{solution}
$41$.

Removing any edge of $C_{41}$ results in a spanning tree.
\end{solution}

\inbook{
\ppart How many spanning trees does $K_{41}$ have?

\begin{center}
\exambox{0.7in}{0.5in}{-0.1in}
\end{center}

\begin{solution}
$41^{39}$.  See Problem~(\bref{CP_numbered_trees}).
\end{solution}
}

\ppart\label{numl11p} How many length-10 paths are there in $K_{41}$?
\inhandout{\emph{Reminder}: A length-10 path has 11 vertices.}

\begin{center}
\exambox{0.7in}{0.5in}{-0.1in}
\end{center}

\begin{solution}
\[
\frac{41!}{30!} = 41 \cdot 40 \cdots 32 \cdot 31.
\]
There is a bijection between length-10 paths and length-11 sequences
of distinct vertices.
\end{solution}

\inbook{
\ppart How many cycles are there in $K_{41}$?  \inhandout{Reminder: in
  a simple graph, cycles must include at least three vertices.}

\begin{center}
\exambox{2.0in}{0.5in}{-0.1in}
\end{center}

\begin{solution}
\[
2^{41} - 1 - 41 - \binom{41}{2}.
\]

This is the number of subsets of $\Zintvco{0}{41}$ that are not of
size zero, one or two.
\end{solution}
}

\eparts

\end{problem}

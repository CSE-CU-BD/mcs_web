\begin{problem}
Prove by Induction:
\begin{equation}
  \sum_{i=0}^{n} i^3  = {\frac{n(n+1)}{2}}^{2}, \forall n \geq 0
\end{equation}
using the equation itself as the induction hypothesis, $P(n)$.

\bparts

\ppart Prove the \inductioncase{base case} $(n=0)$.

\begin{solution}
  \[
{\frac{(0)(1)}{2}}^2 = 0
\]
Therefore, $P(0)$ is true.
\end{solution}

\examspace[2in]

\ppart Now prove the \inductioncase{inductive step}.

\begin{solution}

Let $P(n) ::= (\sum_{i=0}^{n} i^3 = {\frac{n(n+1)}{2}}^{2})$.

Suppose $P(n)$ is true, we need to show $P(n+1)$.

\begin{proof}
	$\sum_{i=0}^{n} i^3 = {\frac{n(n+1)}{2}}^{2}$

	$\leftrightarrow (\sum_{i=0}^{n} i^3) + (n+1)^3 = {\frac{n(n+1)}{2}}^{2} + (n+1)^3$
	
	$\leftrightarrow \sum_{i=0}^{n+1} i^3 = (n+1)^2 ( {\frac{n}{2}}^{2}+ (n+1))$
	
	$\leftrightarrow \sum_{i=0}^{n+1} i^3 = (n+1)^2 ( {\frac{n^2}{4}}+ \frac{(4n+4)}{4})$
	
	$\leftrightarrow \sum_{i=0}^{n+1} i^3 = (n+1)^2 ( {\frac{n^2 + 4n + 4}{4}})$

	$\leftrightarrow \sum_{i=0}^{n+1} i^3 = (n+1)^2 ( {\frac{(n+2)^2}{4}})$
	
	$\leftrightarrow \sum_{i=0}^{n+1} i^3 = {\frac{(n+1)(n+2)}{2}}^{2}$

	$\leftrightarrow P(n+1) \\

	Therefore, $\sum_{i=1}^{n} i^3  = {\frac{n(n+1)}{2}}^{2}, \forall n \geq 0$.
\end{proof}

\end{solution}

\eparts

\end{problem}

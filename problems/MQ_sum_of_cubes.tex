\begin{problem}
  Prove by Induction:
  \begin{equation}
  	\sum_{i=1}^{n} i^3  = {\frac{n(n+1)}{2}}^{2}, \forall n \in N
  \end{equation}
  	Let's prove it by induction:
	Base case: $n=1$:
	$$LHS = 1$$
	$$RHS = {\frac{2}{2}}^2 = 1$$
	Thus, LHS (left hand side) and RHS (right hand side) are equal and it holds for $n=1$
	
	Let $P(n)::= \sum_{i=0}^{n} i^3$. Suppose that the statement $P(n)={\frac{n(n+1)}{2}}^{2}$ for $n$ let's prove that $P(n+1) = {\frac{(n+1)(n+2)}{2}}^{2}.
	
	$$P(n+1) =  \sum_{i=0}^{n+1} i^3 =  \sum_{i=0}^{n} i^3 + (n+1)^3$$
	
	$$\leftrightarrow P(n+1) = P(n) + (n+1)^3$$
	
	Using the hypothesis that  $P(n)={\frac{n(n+1)}{2}}^{2}$, then:
	
	$$\leftrightarrow P(n+1) =  {\frac{n(n+1)}{2}}^{2}+ (n+1)^3$$
	
	$$\leftrightarrow P(n+1) = (n+1)^2 ( {\frac{n}{2}}^{2}+ (n+1))$$
	
	$$\leftrightarrow P(n+1) = (n+1)^2 ( {\frac{n^2}{4}}{2}+ \frac{(4n+4)}{4})$$
	
	$$\leftrightarrow P(n+1) = (n+1)^2 ( {\frac{n^2 + 4n + 4}{4}}{2})$$
	
	$$\leftrightarrow P(n+1) = (n+1)^2 ( {\frac{(n+2)^2}{4}}{2})$$
	
	$$\leftrightarrow P(n+1) = {\frac{(n+1)(n+2)}{2}}^{2}$$

	Therefore, $\sum_{i=1}^{n} i^3  = {\frac{n(n+1)}{2}}^{2}, \forall n \in N$
\end{problem}

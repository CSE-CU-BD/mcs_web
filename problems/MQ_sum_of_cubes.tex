\begin{problem}
Prove by induction:
\begin{equation}
  \sum_{i=0}^{n} i^3  = {\left ( \frac{n(n+1)}{2} \right )}^{2}, \forall n \geq 0
\end{equation}
using the equation itself as the induction hypothesis, $P(n)$.

\bparts

\ppart Prove the \inductioncase{base case} $(n=0)$.

\begin{solution}
  \[
{\frac{(0)(1)}{2}}^2 = 0
\]
Therefore, $P(0)$ is true.
\end{solution}

\examspace[2in]

\ppart Now prove the \inductioncase{inductive step}.

\begin{solution}

Let $P(n) ::= \left ( \sum_{i=0}^{n} i^3 = {\frac{n(n+1)}{2}}^{2} \right )$.

Suppose $P(n)$ is true, we need to show $P(n+1)$.

\begin{proof}
	$\sum_{i=0}^{n} i^3 = {\left ( \frac{n(n+1)}{2} \right )}^{2}$

	$\leftrightarrow {\left ( \sum_{i=0}^{n} i^3 \right )} + (n+1)^3 = {\frac{n(n+1)}{2}}^{2} + (n+1)^3$
	
	$\leftrightarrow \sum_{i=0}^{n+1} i^3 = (n+1)^2 {\left ( {\left ( \frac{n}{2} \right )}^{2}+ (n+1) \right )}$
	
	$\leftrightarrow \sum_{i=0}^{n+1} i^3 = (n+1)^2 {\left ( {\frac{n^2}{4}}+ \frac{4n+4}{4} \right )}$
	
	$\leftrightarrow \sum_{i=0}^{n+1} i^3 = (n+1)^2 {\left ( {\frac{n^2 + 4n + 4}{4}} \right )}$

	$\leftrightarrow \sum_{i=0}^{n+1} i^3 = (n+1)^2 {\left ( {\frac{(n+2)^2}{4}} \right )}$
	
	$\leftrightarrow \sum_{i=0}^{n+1} i^3 = {\left ( \frac{(n+1)(n+2)}{2} \right )}^{2}$

	$\leftrightarrow P(n+1)$ \\

	Therefore, $\sum_{i=1}^{n} i^3  = {\left ( \frac{n(n+1)}{2} \right )}^{2}, \forall n \geq 0$.
\end{proof}

\end{solution}

\eparts

\end{problem}

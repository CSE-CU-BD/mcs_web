\begin{problem}

\newcommand{\spc}{\hspace{0.5in}}
\newcommand{\apart}[1]{\vspace{0.2in}
    \ppart[1]\ \ \parbox[t]{4.5in}{#1}\hfill\brule{0.75in}}

Here are the solutions to the next 10 problem parts, in no particular
order.

\[
n^m \spc m^n \spc \frac{n!}{(n-m)!} \spc
\binom{n-1+m}{m} \spc \binom{n-1+m}{n} \spc 2^{mn}
\]

\bparts

\apart{How many ways are there to put $m$ indistinguishable
balls into $n$ distinguishable urns?}

\begin{solution}
\[
\binom{n - 1 + m}{m}
\]
This is the number of sequences with $m$ stars and $n - 1$ bars.
\end{solution}

\apart{How many ways are there to put $m$ distinguishable balls into
$n$ distinguishable urns?}

\begin{solution}
\[
\frac{n!}{(n-m)!}
\]
that is, $P(n,m)$.
\end{solution}

\apart{How many solutions over the natural numbers are there to the
equation $x_1 + x_2 + \ldots + x_n = m$?}

\begin{solution}
\[
\binom{n - 1 + m}{m}
\]
This is the number of sequences with $m$ stars and $n - 1$ bars.
\end{solution}


\apart{How many $\sqrt{n} \times \sqrt{n}$ matrices are there with
entries drawn from $\set{1, 2, \ldots, m}$?}

\begin{solution}
$m^n$
\end{solution}


\apart{How many different subsets of the set $A \times B$ are there,
if $|A| = m$ and $|B| = n$?}

\begin{solution}
$2^{mn}$
\end{solution}

\apart{How many functions are there from set $A$ to set $B$, if $|A| =
n$ and $|B| = m$?}

\begin{solution}
$m^n$
\end{solution}

\apart{How many $m$-letter words can be formed from an $n$-letter \\
alphabet, if no letter is used more than once?}

\begin{solution}
\[
\frac{n!}{(n-m)!}
\]
that is, $P(n,m)$.
\end{solution}

\apart{How many $m$-letter words can be formed from an $n$-letter \\
alphabet, if letters can be reused?}

\begin{solution}
$n^m$
\end{solution}

\apart{How many relations are there from set $A$ to set $B$, where
$|A| = m$ and $|B| = n$?}

\begin{solution}
$2^{mn}$
\end{solution}

\apart{How many injections are there from set $A$ to set $B$, where
$|A| = m$ and $|B| = n$?}

\begin{solution}
\[
\frac{n!}{(n-m)!}
\]
that is, $P(n,m)$.
\end{solution}

\eparts

\end{problem}

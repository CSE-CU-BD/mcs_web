\newcommand{\spc}{\hspace{0.5in}}
\newcommand{\apart}[1]{\vspace{0.2in}

Here are the solutions to the next 10 problem parts, in no particular
order.

\[
n^m \spc m^n \spc \frac{n!}{(n-m)!} \spc
\binom{n+m}{m} \spc \binom{n-1+m}{m} \spc \binom{n-1+m}{n} \spc 2^{mn}
\]

\bparts

\apart{How many solutions over the natural numbers are there to the
inequality $x_1 + x_2 + \ldots + x_n \leq m$?}

\begin{solution}
\[
\binom{n + m}{m}
\]
This is the same as the number of solutions to the equation the equality
$x_1 + x_2 + \ldots + x_n + y = m$, and which has a bijection to sequences
with $m$ stars and $n$ bars.
\end{solution}

\iffalse

\apart{How many $\sqrt{n} \times \sqrt{n}$ matrices are there with
entries drawn from $\set{1, 2, \ldots, m}$?}

\begin{solution}
$m^n$
\end{solution}

\apart{How many different subsets of the set $A \times B$ are there,
if $\card{A} = m$ and $\card{B} = n$?}

\begin{solution}
$2^{mn}$
\end{solution}


\apart{How many functions are there from set $A$ to set $B$, if $\card{A} =
n$ and $\card{B} = m$?}

\begin{solution}
$m^n$
\end{solution}
\fi

\apart{How many length $m$ words can be formed from an $n$-letter \\
alphabet, if no letter is used more than once?}

\begin{solution}
\[
\frac{n!}{(n-m)!}
\]
%that is, $P(n,m)$.
\end{solution}

\apart{How many length $m$ words can be formed from an $n$-letter \\
  alphabet, if letters can be reused?}

\begin{solution}
$n^m$
\end{solution}

\apart{How many relations are there from set $A$ to set $B$ when
$\card{A} = m$ and $\card{B} = n$?}

\begin{solution}
$2^{mn}$
\end{solution}

\apart{How many injections are there from set $A$ to set $B$, where
$\card{A} = m$ and $\card{B} = n \geq m$?}

\begin{solution}
\[
\frac{n!}{(n-m)!}
\]
%that is, $P(n,m)$.
\end{solution}

\iffalse

\apart{
How many different subsets of the set $A \times B$ are there
when $\card{A} = m$ and $\card{B} = n$?
}
\begin{solution}
$2^{mn}$
\end{solution}
\fi

\apart{How many ways are there to place a total of $m$ distinguishable
  balls into $n$ distinguishable urns, with some urns possibly empty or
  with several balls?}

\begin{solution}
\[
n^m
\]
\end{solution}

\apart{How many ways are there to place a total of $m$ indistinguishable
  balls into $n$ distinguishable urns, with some urns possibly empty or
  with several balls?}

\begin{solution}
\[
\binom{n - 1 + m}{m}
\]
This is the number of sequences with $m$ stars and $n - 1$ bars.
\end{solution}

\apart{How many ways are there to put a total of $m$ distinguishable balls
  into $n$ distinguishable urns with exactly one ball in each urn?}

\begin{solution}
\[
\frac{n!}{(n-m)!}
\]
%that is, $P(n,m)$.
\end{solution}

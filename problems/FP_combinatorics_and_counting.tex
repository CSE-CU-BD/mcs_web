\begin{problem} \textbf{Combinatorics and Counting}

\newcommand{\spc}{\hspace{0.5in}}

\begin{problemparts}

\problempart
Here are the solutions to the next 10 counting questions, in no particular
order:
\[
n^m \spc m^n \spc \frac{n!}{(n-m)!} \spc
\binom{n-1+m}{m} \spc \binom{n-1+m}{n} \spc 2^{mn}
\]

\vspace{.3in}

\begin{tabular}{rll}
1. & \parbox[t]{5in}{
How many ways are there to put $m$ indistinguishable balls into $n$ 
distinguishable urns?
} & \instatements{\brule{0.5in}}
    \insolutions{$\binom{n - 1 + m}{m}$} \\
\\
2. & \parbox[t]{5in}{
How many ways are there to put $m$ distinguishable balls into $n$ 
distinguishable urns?
} & \instatements{\brule{0.5in}}
    \insolutions{$n^m$} \\
\\
3. & \parbox[t]{5in}{
How many solutions over the nonnegative integers are there to the
equation $x_1 + x_2 + \ldots + x_m = n$?
} & \instatements{\brule{0.5in}}
    \insolutions{$\binom{n - 1 + m}{n}$} \\
\\
4. & \parbox[t]{5in}{
How many $\sqrt{n} \times \sqrt{n}$ matrices are there with
entries drawn from $\set{1, 2, \ldots, m}$?
} & \instatements{\brule{0.5in}}
    \insolutions{$m^n$} \\
\\
5. & \parbox[t]{5in}{
How many different subsets of the set $A \times B$ are there,
if $|A| = m$ and $|B| = n$?
} & \instatements{\brule{0.5in}}
    \insolutions{$2^{mn}$} \\
\\
6. & \parbox[t]{5in}{
How many total functions are there from set $A$ to set $B$, 
if $|A| = n$ and $|B| = m$?
} & \instatements{\brule{0.5in}}
    \insolutions{$m^n$} \\
\\
7. & \parbox[t]{5in}{
How many $m$-letter words can be formed from an $n$-letter
alphabet, if no letter is used more than once? Assume $n \geq m$.
} & \instatements{\brule{0.5in}}
    \insolutions{$\frac{n!}{(n-m)!}$} \\
\\
8. & \parbox[t]{5in}{
How many $m$-letter words can be formed from an $n$-letter
alphabet, if letters can be reused?
} & \instatements{\brule{0.5in}}
    \insolutions{$n^m$} \\
\\
9. & \parbox[t]{5in}{
How many relations are there from set $A$ to set $B$, where
$|A| = m$ and $|B| = n$?
} & \instatements{\brule{0.5in}}
    \insolutions{$2^{mn}$} \\
\\
10. & \parbox[t]{5in}{
How many total injective functions are there from set $A$ to 
set $B$, where $|A| = m$ and $|B| = n \geq m$?
} & \instatements{\brule{0.5in}}
    \insolutions{$\frac{n!}{(n-m)!}$} \\
\\
\end{tabular}

\end{problemparts}
\end{problem}

\begin{problem}
For each pair $(A,B)$ of sets below, indicate ALL of the following
kinds of relations between them that are possible.  The kinds of relations are:

\begin{enumerate}
\item TI:     Total Injection
\item TIF:     Total injective Function
\item SF:     Surjective Function
\item TSF:     Total Surjective Function
\end{enumerate}
       
The pairs are:

\begin{enumerate}

\item (\integers, \naturals)

\item (\pow(\rationals), \naturals)

\item ($\reals \times \reals$, $\reals$)

\item (\text{infinite binary string} (\binw),\text{ infinite strings of }$\{2,3,4\}$)

\item (integers in $[0,2^{100}]$, \power(\set{$0,1,...,100$}))

\item (\emptyset, \integers)

\item (\naturals, \emptyset) 
\end{enumerate}


\begin{solution}

\begin{enumerate}
\item ALL
\item SF, TSF
\item ALL
\item ALL
\item TI, TIF
\item TI, TIF
\item SF
\end{enumerate}

\end{problem}

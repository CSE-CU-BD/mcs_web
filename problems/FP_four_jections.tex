\begin{problem}
For each pair $(A,B)$ of sets below, indicate ALL of the following
kinds of relations between them that are possible.  The kinds of
relations are:

\begin{enumerate}
\item TI:     Total Injection
\item TIF:    Total injective Function
\item SF:     Surjective Function
\item TSF:    Total Surjective Function
\end{enumerate}

\begin{center}
\begin{tabular}{rl}
 TI:   &    Total Injection\\
 TIF:  &  Total injective Function\\
 SF:   & Surjective Function\\
 TSF:  &  Total Surjective Function
\end{tabular}
\end{center}
      
The pairs are:

\begin{enumerate}[i]

\item $(\integers\quad \nngint)$\hfill\examrule

\item $(\power(\rationals)\quad \nngint)$\hfill\examrule

\item $(\reals \times \reals\quad \reals)$\hfill\examrule

\item $(\text{infinite binary string}\ (\binw)\quad\ \text{infinite strings
  of}\ \set{2,3,4} (\infstrings{2,3,4}))$\hfill\examrule

\item $(\text{The integer interval}\ \Zintv{0}{2^{100}}\quad \power(\set{0,1,...,100})$\hfill\examrule

\item $(\emptyset\quad \integers)$\hfill\examrule

\item $(\nngint\quad \emptyset)$\hfill\examrule
\end{enumerate}

\begin{solution}

\begin{enumerate}[i]
\item ALL
\item SF, TSF
\item ALL
\item ALL
\item TI, TIF
\item TI, TIF
\item SF
\end{enumerate}

\end{solution}

\end{problem}

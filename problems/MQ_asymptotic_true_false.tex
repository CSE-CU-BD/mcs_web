\begin{problem}
Circle each of the true statements below. 

Explanations are not required, but partial credit for wrong answers will not be given without them. 

\begin{itemize}

\item $\displaystyle n^2 \sim n^{2} + n$

\item $\displaystyle 3^n=O\left(2^n\right)$

\item $\displaystyle n^{\sin (n\pi/2) + 1} = o\left(n^2\right)$

%\item $\displaystyle \ln \left(\left(n^2\right)!\right) = O\left(n^2\ln(n)\right)$ 

\item $\displaystyle n = \Theta \left(\frac{3n^3}{(n+1)(n-1)}\right)$

\end{itemize}

\begin{solution}The 1st and 4th statements are true.

\begin{itemize}

\item $\frac{n^{2} + n}{n^2} = \frac{n^2}{n^2} + \frac{n}{n^2}$. The ratio approaches $1 + 0 = 1$, 
therefore the two expressions are similar.

\item $\lim_{n\to\infty}{\frac{3^n}{2^n}} = \lim_{n\to\infty}{(3/2)^n} = \infty$

\item while the left side never exceed $n^2$, when n is 1,5,9,13,..., the left side is equal to $n^2$ 
which means it's not little-o of $n^2$.

\item $3n^3 = \Theta \left(n^3\right)$. $(n+1)(n-1) = \Theta \left(n^2\right)$. 
Therefore $\frac{3n^3}{(n+1)(n-1)} = \Theta \left(n\right)$.

\end{itemize}

\end{solution}

\end{problem}

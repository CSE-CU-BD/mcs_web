\begin{problem}
Two people are born on two distinct days, say $(d_1,d_2)$. Assume 365 days.
\bparts

\ppart How many possible $(d_1,d_2)$ pairs are there?

\begin{solution}
$365*364$
\end{solution}

\ppart Now suppose the birthdays are not necessarily  distinct. How many possible $(d_1,d_2)$ pairs are there?

\begin{solution}
$365^2$
\end{solution}

\eparts

\end{problem}

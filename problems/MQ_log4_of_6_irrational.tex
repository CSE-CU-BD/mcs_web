\begin{problem}
 Prove that $\log_4 6$ is irrational.

\begin{solution}
\begin{proof}
  Suppose to the contrary that $\log_4 6$ is rational, so $\log_4 6 =
  m/n$ for some positive integers $m$ and $n$.  So $m = n\log_4 6$.
  Now raising 4 to each side of this equation gives
\begin{equation}\label{4m4n6}
4^m = 4^{n \log_4 6} =  (4^{\log_4 6})^n = 6^n.
\end{equation}
But this is impossible, since right hand side of~\eqref{4m4n6} is
divisible by 3 and the left hand side is not.

This contradiction implies that $\log_4 6$ must be irrational.
\end{proof}
\end{solution}
\end{problem}

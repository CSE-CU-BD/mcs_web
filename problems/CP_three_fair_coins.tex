\begin{problem}
Suppose that you flip three fair, mutually independent coins.
Define the following events:
\begin{itemize}
\item Let $A$ be the event that \emph{the first} coin is heads.
\item Let $B$ be the event that \emph{the second} coin is heads.
\item Let $C$ be the event that \emph{the third} coin is heads.
\item Let $D$ be the event that \emph{an even number of} coins are heads.
\end{itemize}

\bparts

\ppart Use the four step method to determine the probability space for
this experiment and the probability of each of $A,B,C,D$.

\begin{solution}
The tree is a binary tree with depth 3 and 8 leaves.  The
  successive levels branching to show whether or not the successive events
  $A,B,C$ occur.  By definition of \emph{fair} and \emph{independent},
  each branch out of a vertex is equally likely to be followed.  So the
  probability space has as outcomes the eight length-3 strings of
  $\mathtt{H}$'s and $\mathtt{T}$'s, each of which has probability
  $(1/2)^3 = 1/8$.

  Each of the events events $A,B,C,D$ are true in four of the outcomes
  and hence has probability 1/2.
\end{solution}

\ppart Show that these events are not mutually independent.

\begin{solution}
\[
\pr{A \cap B \cap C \cap D} = 0 \neq (1/2)^4 = \pr{A} \cdot \pr{B} \cdot
\pr{C} \cdot \pr{D}.
\]
\end{solution}

\ppart Show that they are 3-way independent.

\begin{solution}
Because the coin tosses are mutually independent, we know:
\[
\pr{A \cap B \cap C} = \pr{A} \cdot \pr{B} \cdot \pr{C}.
\]
What remains is to check that equality holds for the other subsets of
three events: $\set{A,B,D}$, $\set{A,C,D}$, and $\set{B,C,D}$.  By
symmetry, again, we need only check one, say the first one.
\[
\pr{A \cap B \cap D} = \pr{\set{HHT}} = \dfrac{1}{8}.
\]
Since this is equal to $\pr{A} \cdot \pr{B} \cdot \pr{D}$, these three
events are independent.

We conclude that all four events are three-way independent.
\end{solution}

\eparts                                                                                
\end{problem}

each line of the following proof \instatements{(without  looking it up in Theorem~\ref{th:prod})} that if $R_1$ and $R_2$ are\emph{independent}, then\[\expect{R_1 \cdot R_2} = \expect{R_1} \cdot \expect{R_2}.\]\begin{proof}\begin{align*}\lefteqn{\expect{R_1 \cdot R_2}}\\ & =  \sum_{r \in \range{R_1\cdot R_2}} r\cdot \pr{R_1\cdot R_2=r}\\ & =  \sum_{r_i \in \range{R_i}}            r_1 r_2 \cdot \pr{R_1=r_1\ \text{ and }\ R_2=r_2}\\ & =  \sum_{r_1 \in \range{R_1}} \sum_{r_2 \in \range{R_2}}            r_1 r_2 \cdot \pr{R_1=r_1\ \text{ and }\ R_2=r_2}\\ & =  \sum_{r_1 \in \range{R_1}} \sum_{r_2 \in \range{R_2}}            r_1 r_2 \cdot \pr{R_1=r_1}\cdot \pr{R_2=r_2}\\ & =  \sum_{r_1 \in \range{R_1}} \paren{r_1\pr{R_1=r_1} \cdot              \sum_{r_2 \in \range{R_2}} r_2 \pr{R_2=r_2}}\\ & =  \sum_{r_1 \in \range{R_1}} r_1\pr{R_1=r_1} \cdot \expect{R_2}\\ & =   \expect{R_2} \cdot \sum_{r_1 \in \range{R_1}} r_1\pr{R_1=r_1}\\ & =   \expect{R_2} \cdot  \expect{R_1}.\end{align*}\end{proof}\begin{solution}\begin{proof}\begin{align*}\lefteqn{\expect{R_1 \cdot R_2}}\\ & \eqdef \sum_{r \in \range{R_1\cdot R_2}} r\cdot \pr{R_1\cdot R_2=r} & \text{(by definition)}\\ & =  \sum_{r_i \in \range{R_i}}            r_1 r_2 \cdot \pr{R_1=r_1 \QAND R_2=r_2} & \text{(event $[R_1 \cdot R_2=r]$ splits into events}\\   &&\quad [R_1 = r_1 \QAND R_2 = r_2] \text{ such that }r_1 r_2 = r)\\ & =  \sum_{r_1 \in \range{R_1}} \sum_{r_2 \in \range{R_2}}            r_1 r_2 \cdot \pr{R_1=r_1 \QAND R_2=r_2}     &\text{    (ordering terms in the sum)}\\ & =  \sum_{r_1 \in \range{R_1}} \sum_{r_2 \in \range{R_2}}            r_1 r_2 \cdot \pr{R_1=r_1}\cdot \pr{R_2=r_2}                    &\text{    (independence of $R_1,R_2$)}\\ & =  \sum_{r_1 \in \range{R_1}} \paren{r_1\pr{R_1=r_1} \cdot              \sum_{r_2 \in \range{R_2}} r_2 \pr{R_2=r_2}}                    &    \text{(factor out $r_1\pr{R_1=r_1}$)}\\ & =  \sum_{r_1 \in \range{R_1}} r_1\pr{R_1=r_1} \cdot \expect{R_2}                    &    \text{(def of $\expect{R_2}$)}\\ & =  \expect{R_2} \cdot \sum_{r_1 \in \range{R_1}} r_1\pr{R_1=r_1}                    &\text{    (factor out $\expect{R_2}$)}\\ & =  \expect{R_2} \cdot  \expect{R_1}.                    &\text{    (def of $\expect{R_1}$)}\end{align*}\end{proof}\end{solution}%\eparts\end{problem}%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% Problem ends

\begin{problem}
What is $\remainder(3^{101},21)$?

\iffalse
\instatements{(No explanation is required, but no part credit without an
  explanation.)}
\fi

\exambox{0.5in}{0.5in}{0.0in}

\begin{solution}
\textbf{12}.

\begin{align*}
\remainder(3^1,21) & = 3\\
\remainder(3^2,21) & = 9\\
\remainder(3^3,21) & = 6\\
\remainder(3^4,21) & = 18\\
\remainder(3^5,21) & = 12\\
\remainder(3^6,21) & = 15\\
\remainder(3^7,21) & = 3.
\end{align*}

So
\begin{align*}
\rem{3^{101}}{21}
  & = \rem{3^{5 + 6 \cdot 16}}{21}\\
  & = \rem{3^4}{21} \cdot \rem{3^{1+6\cdot 16}}{21}\\
  & = 18 \cdot 3\\
  & = 12.
\end{align*}
\end{solution}
\end{problem}

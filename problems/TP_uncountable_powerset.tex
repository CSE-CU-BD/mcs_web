\begin{problem}
Prove that if $S$ is an infinite set, then $\pow{S}$ is uncountable.

\begin{solution}
By Lemma~\bref{smallestinf}, if $\pow{S}$ was countable, then $S \surj
\pow{S}$, contradicting Cantor's Theorem~\bref{powbig}.
\end{solution}

\end{problem}

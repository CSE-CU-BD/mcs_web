\begin{problem}
Use the Law of Total Probability\inbook{~(\bref{multitotalprob})},%
\inhandout{\[
\pr{A} = \sum_{i=1}^{n} \prcond{A}{E_i} \cdot \pr{E_i},
\]
}
to prove the Multi-event Bayes' Rule\inbook{~(\bref{multibayesrule}).}
\inhandout{\[ \prcond{E_1}{A} = \frac{\prcond{A}{E_1}\cdot
    \pr{E_1}}{\sum_{i=1}^{n} \prcond{A}{E_i} \cdot \pr{E_i}}
\]
}

\begin{solution}
\begin{align*}
\prcond{E_1}{A}
  & \eqdef \frac{E_1 \intersect A_1}{\pr{A}}\\
  & =  \frac{E_1 \intersect A_1}{\sum_{i=1}^{n} \prcond{A}{E_i} \cdot \pr{E_i}}
       & \text{(Total probability)}\\
  & =  \frac{\prcond{A}{E_1}\cdot \pr{E_1}}{\sum_{i=1}^{n} \prcond{A}{E_i} \cdot \pr{E_i}}
       & \text{\text{(def conditional prob)}}
\end{align*}

\end{solution}
\end{problem}

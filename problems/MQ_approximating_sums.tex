\begin{problem}

\begin{problemparts}

\problempart
Let $\displaystyle R=\sum_{i=1}^n{\log_2{(x+1)}}$.
Which of the following are suitable bounds for $R$?
\begin{enumerate}
\item $\displaystyle R\leq\int_2^{n+2} \! \log_2{x} \, \mathrm{d}x$  % Good
\item $\displaystyle R\leq \log_2{n}+\int_1^n \! \log_2{(x+1)} \, \mathrm{d}x$  % Bad
\item $\displaystyle R\leq 1+\int_1^n \! \log_2{(x+1)} \, \mathrm{d}x$  % Bad
\item $\displaystyle R\geq 1+\int_1^n \! \log_2{(x+1)} \, \mathrm{d}x$  % Good
\end{enumerate}

\problempart
Let $\displaystyle S=\sum_{i=1}^n{\lceil{x}\rceil}$.
Which of the following are suitable bounds for $S$?
\begin{enumerate}
\item $\displaystyle S\leq\int_1^{n+1} \! \lceil{x}\rceil \, \mathrm{d}x$  % Good
\item $\displaystyle S\leq n +\int_0^{n-1} \! \lceil{x+1}\rceil \, \mathrm{d}x$  % Good
\item $\displaystyle S\leq 1+\int_1^n \! \lceil{x}\rceil \, \mathrm{d}x$  % Good
\item $\displaystyle S\geq 1+\int_1^n \! \lceil{x}\rceil \, \mathrm{d}x$  % Good
\end{enumerate}

\problempart
Let $\displaystyle T=\sum_{i=1}^n{\cos{\frac{\pi x}{100}}}$, where $2\leq n \leq 80$.
Which of the following are suitable bounds for $T$?
\begin{enumerate}
\item $\displaystyle T\leq\int_0^n \! \cos{\frac{\pi x}{100}} \, \mathrm{d}x$  % Good
\item $\displaystyle T\leq  \cos{\frac{\pi}{100}} + \cos{\frac{\pi}{50}} + \int_2^n \! \cos{\frac{\pi x}{100}} \, \mathrm{d}x$  % Good
\item $\displaystyle T\geq \int_1^{n+1} \! \cos{\frac{\pi x}{100}} \, \mathrm{d}x$  % Good
\item $\displaystyle T\geq \cos{\frac{\pi}{100}} + \int_1^n \! \cos{\frac{\pi x}{100}} \, \mathrm{d}x$  % Bad
\end{enumerate}

\problempart
How would your answer to the preceding part change if the only restriction on $n$ was $n\geq 2$?  In a single short sentence, explain why. 

\end{problemparts}
\end{problem}

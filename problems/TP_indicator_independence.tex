\begin{problem}

\bparts

\ppart\label{AindBindbarB} Prove that if $A$ and $B$ are independent events, then so are
$A$ and $\bar{B}$.
\begin{solution}
\begin{proof}
\begin{align*}
\pr{A \intersect \bar{B}}
  &  = \pr{A} - \pr{A \intersect B}
       & \text{(difference rule)}\\
  & =  \pr{A} - \pr{A} \cdot \pr{B}
       & \text{(independence of $A$ and $B$)}\\
  & =  \pr{A} (1 - \pr{B})\\
  & =  \pr{A} \cdot \pr{\bar{B}} & \text{(complement rule)}.
\end{align*}
\end{proof}
\end{solution}

\ppart
Let $I_A$ and $I_B$ be the indicator variables for events $A$ and $B$.
Prove that $I_A$ and $I_B$ are independent iff $A$ and $B$ are
independent.  

\hint For any event, $E$, let $E^1 \eqdef E$ and $E^0 \eqdef \bar{E}$.

\begin{solution}
\begin{proof}
By part~\eqref{AindBindbarB}, $A$ and $B$ are independent iff $\exists
a,b \in \set{0,1}$.[$A^a$ and $B^b$ are independent] iff
$\forall a,b \in \set{0,1}$.[$A^a$ and $B^b$ are independent].

Also, by definition of $I_A$, the event $[I_A = a]$ is the same as $A^a$ and
the event $[I_B = b]$ is the same as $B^b$.  Therefore,

\begin{tabular}{l}
$I_A$ and $I_B$ are independent\\
\quad iff
  $\forall a,b \in \set{0,1}$. [$[I_A = a]$ and $[I_A = b]$ are
  independent events]\\
\qquad\qquad (by def of independence for ran vars),\\
 \quad iff $\forall a,b \in \set{0,1}$. [$A^a$ and $B^b$ are independent]\\
 \quad iff $A$ and $B$ are independent \quad (by part~\eqref{AindBindbarB}).
\end{tabular}

\end{proof}
\end{solution}

\eparts

\end{problem}

\begin{problem}

\bparts

\ppart Write predicates that express the following assertions.

\begin{quote}

\begin{itemize}
\item $R$ is a surjection [$>=1$ in]. \hfill\examrule[1in]
\begin{solution}
$\forall b.\, \exists a.\ a \mrel{R} b$.
\end{solution}

\item $R$ is a function [$<=1$ out]. \hfill\examrule[1in]
\begin{solution}
$\forall b_1,b_2,a.\ (a \mrel{R} b_1 \QAND a \mrel{R} b_2) \QIMPLIES b_1 = b_2$.

$\forall a_1,a_2,b_1,b_2.\ (a_1 \mrel{R} b_1 \QAND a_2 \mrel{R}b_2 \QAND b_1 \neq b_2) \QIMPLIES a_1 \neq a_2$.
\end{solution}

\item $x = y - z$. \hfill\examrule[1in]
\begin{solution}
$\forall w.\, w \in x\ \QIFF\  (w \in y\ \QAND\ \QNOT(w \in z))$.
\end{solution}

\item $x = \{\}$. \hfill\examrule[1in]
\begin{solution}
$\forall z.\, \QNOT(z\in x)$
\end{solution}

\end{itemize}

\end{quote}

\eparts

\end{problem}

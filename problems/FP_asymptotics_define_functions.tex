\begin{problem} \textbf{Asymptotics}
\begin{problemparts} 

\problempart
Define a function $f(n)$ such that $f = \Theta(n^4)$ and $\QNOT (f \sim n^4)$.

f(n) = 

\begin{solution}
2n^4
\end{solution}

\problempart
Define a function $g(n)$ such that $g = O(n^4)$, $g \neq \Theta(n^4)$ and $g \neq o(n^4)$.

g(n) = 

\begin{solution}
n^4\sin(\frac{n\pi}{2}) + n^4\cos(\frac{n\pi}{2})
\end{solution}

\end{problemparts}
\end{problem}

\begin{problem}
Let $D$ be the total number of heads among mutually independent tosses
of $n$ coins whose respective probabilities of coming up heads are
$p,p^2,\dots,p^n$.  What are the expectation and variance of $D$?
\begin{solution}

$D = \sum_1^n H_i$ where $H_i$ is indicator for head on the $i$th toss.
So
\[
\expect{D} = \sum_1^n \expect{H_i} = p \sum{i=0}^{n-1} p^i = \frac{p(1-p^n)}{1-p}.
\]
Variances add because of mutual independence, so
\begin{align*}
\variance{D}
    & = \sum{i=1}^{n} p^i(1-p^i)\\
    & = \sum{i=1}^{n} p^i-p^{2i}\\
    & = p\sum{i=0}^{n} p^i + p^2\paren{\sum{i=0}^{n-1} \paren{p^2}^i}}\\
    & = \frac{p(1-p^n)}{1-p} + \frac{p^2(1-p^{2n})}{1-p^2}.
\end{align*}

\end{solution}
\end{problem}

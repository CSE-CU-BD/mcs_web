\begin{problem}
You have a coin with probability $p$ of flipping heads.  For your
first try, you flip it once.  For your second try, you independently
flip it twice.  You continue until the $n$th try, where you
independently flip it $n$ times.  You \emph{win} a try if you flip all
heads.  Let $W$ be the number of winning tries.  Write a closed-form
expression for $\variance{W}$.

\inhandout{
%$\expect{D_n} =\ $  \exambox{3.0in}{0.6in}{-0.3in}

%$\variance{D_n} =$ 

\begin{center}
\exambox{3.0in}{0.6in}{-0.3in}
\end{center}
}

\begin{solution}
$W = \sum_1^n H_k$ where $H_k$ is indicator winning the $k$th try.

Since $\pr{H_k=1} = p^k$, we have $\variance{H_k} = p^k(1-p^k)$.
\iffalse
So
\[
\expect{D_n} = \sum_{i=1}^n \expect{H_i} = \sum_{i=1}^n p^i = p \sum_{i=0}^{n-1} p^i = \frac{p(1-p^n)}{1-p}.
\]
\fi

Variances add because of mutual independence, so
\begin{align*}
\variance{W}
    & = \sum_{k=1}^{n} p^k(1-p^k)\\
    & = \sum_{k=1}^{n} p^k-p^{2k}\\
    & = p\sum_{k=0}^{n-1} p^k - p^2\paren{\sum_{k=0}^{n-1} \paren{p^2}^k}\\
    & = p\frac{1-p^n}{1-p} - p^2\frac{1-p^{2n}}{1-p^2}.
\end{align*}

\end{solution}

\begin{editingnotes}
Could make Chebyshev, Chernoff, $\cdf$ estimate comparison.
\end{editingnotes}

\end{problem}

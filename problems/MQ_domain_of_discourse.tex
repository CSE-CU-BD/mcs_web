\begin{problem}
For each of the logic formulas below, indicate the smallest domain in
which it is true, among
\[
 \nngint\ \text{(nonnegative integers)}, \integers\ \text{(integers)},
 \rationals\ \text{(rationals)}, \reals\ \text{(reals)}, \complexes\
 \text{(complex numbers)},
\]
or state ``\textbf{none}'' if it is not true in any of them. Please briefly explain each one.

\begin{enumerate}[i.]
\item $\forall x \exists y.\, y = 3x$                      \hfill\examrule
  \begin{solution}
    $\nngint$. Can always multiply $x$ by $3$.
  \end{solution}

  \examspace[1in]
  
\item $\forall x \exists y.\, 3y = x$                      \hfill\examrule
  \begin{solution}
    $\rationals$. Must be able to divide by $3$.
  \end{solution}

  \examspace[1in]

\item $\forall x \exists y.\, y^2 = x$                     \hfill\examrule
  \begin{solution}
    $\complexes$. Every number must have a square root.
  \end{solution}

  \examspace[1in]


\item $\forall x \exists y.\, y < x$                       \hfill\examrule
  \begin{solution}
    $\integers$. Can always subtract $1$ in $\integers$. In $\nngint$, nothing is smaller than $x=0$.
  \end{solution}
  
  \examspace[1in]


\item $\forall x \exists y.\, y^3 = x$                     \hfill\examrule
  \begin{solution}
    $\reals$.  Every real number does have a real cube root.
  \end{solution}

  \examspace[1in]

\item $\forall x\neq 0.\, \exists y,z.\, y \neq z\QAND\  y^2 = x = z^2$  \hfill\examrule
  \begin{solution}
     $\complexes$.  Every nonzero has 2 distinct square roots.
  \end{solution}

\iffalse
\item $\forall x\neq 0.\, \exists y .\, y \neq x \QAND\ y^2 = x^2$    \hfill\examrule
  \begin{solution}
    Every nonzero has 2 sqrts: $\reals$
  \end{solution}
\fi

\end{enumerate}

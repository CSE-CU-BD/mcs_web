\begin{problem}

\textbf{Combinatorics and Counting}

\bparts

\ppart
Suppose that we are rolling a fair 6-sided dice $n$ times. What is 
the probability that $k$ consecutive results turn out to be the same?
(e.g. there are \textit{7} consecutive \textit{2}s in the result: 
\textit{512322222221}.)

\begin{solution}
  Consider the possible results as an $n$ length string with characters
  ${1,2,3,4,5,6}$. The total of all possible results is clearly $6^n$.
  
  We can then count the total number of results that have $k$ length
  substrings made of the same character: there are $6$ possible characters 
  to choose from; $n - k + 1$ places to begin the $k$ length substring; 
  and $6^{(n-k)}$ ways to fill in remaining $n-k$ results. 
  There are, therefore, a total of $6 \cdot (n-k+1) \cdot 6^{(n-k)}$
  number of possible strings.
  
  The probability is the number of results with $k$ consecutive results 
  divided by the number of all possible results:
  
  \[ 6 \cdot (n-k+1) \cdot 6^{(n-k)} \over 6^n \]
\end{solution}

\ppart
Suppose that we are rolling a fair 6-sided dice $n$ times. What is 
the probability that there are exactly $k$ \textit{1}s, where none of 
the \textit{1}s follow another \textit{1} immediately?

\begin{solution}
  There are $\binom{n-k+1}{k}$ number of ways to choose where to 
  place the \textit{1}s (consider a bijection with having $k$ 1s 
  in an $n-k+1$ bit binary string). The remaining characters can only
  choose from the characters ${2,3,4,5,6}$, so there are only
  $5^{(n-k)}$ ways to choose the remaining characters.
  
  Similar to the previous part, the total number of results is $6^n$, so
  the probability is:
  
  \[ \binom{n-k+1}{k} \cdot 5^{(n-k)} \over {6^n} \]
\end{solution}

\eparts
\end{problem}

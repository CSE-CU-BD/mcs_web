\ppart Let $R$ be a binary relation on a set $D$.  Let $x,y$ be variables
ranging over $D$.  \inbook{Indicate}\inhandout{Circle} the expressions
below whose meaning is that $R$ is an \emph{injection} $[\leq
  1\ \text{in}]$.  Remember $R$ is a not necessarily total or a
function.

\begin{enumerate}[(i)]
%\item $R(x) \intersect R(y) = \emptyset$
\item $R(x) = R(y) \QIMPLIES x = y$
\item $R(x) \intersect R(y) = \emptyset \QIMPLIES x \neq y$
\item $R(x) \intersect R(y) \neq \emptyset \QIMPLIES x \neq y$
\item\label{RxintRyneq} $R(x) \intersect R(y) \neq \emptyset \QIMPLIES x = y$
\item $\inv{R}(R(x)) = \set{x}$
\item\label{invRRsubsx} $\inv{R}(R(x)) \subseteq \set{x}$
\item $\inv{R}(R(x)) \supseteq \set{x}$
\item $R(\inv{R}(x)) = x$
%\item $\QNOT(\exists x.\, \exists y \neq x.\, R(x) \intersect R(y) \neq \emptyset)$
\end{enumerate}

\begin{solution}
Only~\ref{RxintRyneq} and~\ref{invRRsubsx}.
\end{solution}

\ppart Give an example of a set $S$ such that there is no injection
from $S$ to the real interval $[0,1]$.

\begin{solution}
$\power(\reals)$.
\end{solution}

\eparts

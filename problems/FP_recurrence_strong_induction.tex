\begin{problem}
We define the sequence of numbers
\[
a_n = \begin{cases}
  a_{n-1} + a_{n-2} + a_{n-3}+ a_{n-4}
      &  \text{if $n \geq 4$,}\\
  1   &  \text{if $0 \leq n \leq 3$.}\\
 \end{cases}
\]

Prove that $a_n \equiv 1 \pmod{3}$ for all $n\geq 0$. 

\begin{solution}
Proof by strong induction with induction hypothesis
\[
P(n)\eqdef\quad a_n \equiv 1 \pmod{3}.
\]

\inductioncase{Base case} ($0\leq n\leq 3$):  $a_n=1$ and is therefore 
$\equiv 1 \pmod{3}$, so $P(n)$ holds.

\inductioncase{Inductive step}: For $n \geq 3$, assume $P(k)$ for $0\leq k < n$ 
in order to prove $P(n)$.

In particular, $P(k)$ holds for $k = n-4,n-3,n-2$ and $n-1$, each of
$a_{n-4}$, $a_{n-3}$, $a_{n-2}$ and $a_{n-1}$ is $\equiv 1 \pmod{3}$,
so their sum, $a_n$, is $ \equiv 1+1+1+1 = 4 \equiv 1 \pmod{3}$.  That
is, $a_n \equiv 1 \pmod{3}$, which means $P(n)$ holds.
\end{solution}

\end{problem}

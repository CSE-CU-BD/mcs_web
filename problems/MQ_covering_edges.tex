\newcommand{\covering}[1]{\text{covering}\paren{#1}}

\begin{problem}
  
  If $a$ and $b$ are distinct vertices of a digraph, then $a$ is said
  to \term{cover} $b$ if there is an edge from $a$ to $b$ and every
  path from $a$ to $b$ traverses this edge.  If $a$ covers $b$, the
  edge from $a$ to $b$ is called a \term{covering edge}.

  Let $\covering{D}$ be the subgraph of $D$ whose edges are the
  covering edges, and assume $D$ is a finite \idx{DAG}.  It turns out
  that $\covering{D}$ has the same positive path relation as $D$.

This follows from the
\begin{claim*}
If there is a path in $D$ from vertex, $u$, to a different vertex,
$v$, then there is a path from $u$ to $v$ that only traverses covering
edges.
\end{claim*}

  \bparts \ppart Prove the Claim.  \hint Explain why there must be a
  \emph{longest} path from $u$ to $v$.
  
  \examspace[3in]
  
  \begin{solution}
   Since $D$ is a finite DAG, no path can be longer than the number of
   vertices ---otherwise some vertex must repeat on the path, which
   means there would be a cycle.  So there must be a \emph{longest}
   path from $u$ to $v$.  Now every edge on this path must be a
   covering edge or it could be replaced by a path of length 2 or
   more, yielding a longer path from $u$ to $v$.
  \end{solution}
  
  \ppart Show that the Claim fails for a finite digraph, $G$, with three
  vertices and an edge from every vertex to each of the other two vertices.
  \hint What are the covering edges of $G$?

\begin{solution}
  There are no covering edges in $G$, since for each edge
  $\diredge{u}{v}$ there is a length 2 path $uwv$ through the
  remaining vertex, $w$, that does not traverse this edge.  So the
  positive path relation of $\covering{G}$ is empty, even though there
  is a positive length path between every pair of vertices, including
  from each vertex to itself.
\end{solution}
\eparts

\end{problem}

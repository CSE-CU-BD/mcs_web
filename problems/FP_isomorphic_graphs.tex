\begin{problem}
For each of the following pairs of graphs, either define an isomorphism
between them, or prove that there is none.  (We write $ab$ as shorthand
for $\edge{a}{b}$.)

\bparts

\ppart
\begin{align*}
G_1\text{ with }& V_1 = \set{1,2,3,4,5,6},\ E_1= \set{12, 23, 34, 45, 56, 61, 52} \\
G_2\text{ with }& V_2 = \set{a,b,c,d,e,f},\ E_2= \set{ab, bc, cd, da, be, ef, fc} \\
\end{align*}

\begin{solution}
Isomorphic. abefdc.
\end{solution}

\ppart
\begin{align*}
G_1\text{ with }& V_1 = \set{1,2,3,4,5,6},\ E_1=  \set{13, 35, 57, 74, 46, 61, 43, 32} \\
G_2\text{ with }& V_2 = \set{a,b,c,d,e,f},\ E_2=  \set{ab, bc, cd, de, ec, fb, fe, fa} \\
\end{align*}

\begin{solution}
Non-isomorphic, one node in $G_1$ has degree 4.
\end{solution}

\eparts
\end{problem}

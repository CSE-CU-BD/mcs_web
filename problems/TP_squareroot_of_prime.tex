\begin{problem}
Prove that if $a\cdot b =n$, then $a$ or $b$ must be $\le \sqrt{n}$, where
$a,b$, and $n$ are nonnegative integers.
\hint by contradiction

\begin{solution}

\begin{proof}
Suppose to the contrary that $a > \sqrt{n}$ and $b > \sqrt{n}$.  Then
\[
a \cdot b > \sqrt{n} \cdot \sqrt{n} = n,
\]
contradicting the fact that $a\cdot b = n$.
\end{proof}

\end{solution}
\end{problem}

\begin{problem}
\bparts

\ppart Prove that $\log x < x$ for all $x >1$ (requires
elementary calculus).


\ppart
Prove that the relation, $R$, on functions such that $f\mrel{R}g$ iff $g = o(f)$
is a strict partial order.

\ppart
Prove that $f \sim g$ iff $f = g + h$ for some function $h = o(g)$.

\eparts

\end{problem}

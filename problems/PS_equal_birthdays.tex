\begin{problem}
\bparts

\ppart Give an example of mutually independent random variables
$R,S,T$ such that the event $[R=S]$ is not independent of $[R=T]$.

\begin{solution}
Let $R$ and $S$ both take the same constant value.
\end{solution}

\ppart Prove that if $R,S,T$ are mutually independent random variables
and $R$ and $S$ have a uniform distribution, then $[R=S]$ is
independent of $[R=T]$.

\hint Let $p = \pr{R = b}$ and first prove that $\pr{R=T} = p$.

\begin{solution}
\begin{proof}
Since $R$ is uniform, there is a $p$ such that $p = \pr{R = b}$ for
all $b \range{R}$.  Then
\begin{align*}
\pr{R=T}
   & = \pr{\lgunion_{b \in \range{R}} [R=b \QAND\ T=b]\\
   & = \sum_{b \in \range{R}} \pr{[R=b \QAND\ T=b]}
         & \text{(disjoint sum rule)}\\
   & = \sum_{b \in \range{R}} \pr{R=b} \cdot \pr{T=b}
         & (R & T \text{ independent})\\
   & = \sum_{b \in \range{R}} p \cdot \pr{T=b}
          & \text{$R$ is uniform}\\
   & = p \sum_{b \in \range{R}} \pr{T=b} = p.
\end{align*}
Similarly, $\pr{R=S} = p$.  Further, since $S$ is uniform, $\pr{S=T} =
p$.

Now,
\begin{align*}
\pr{R=S \QAND\ R=T}
    & = \lgunion_{b \in \range{R}} [R=b \QAND\ S=b \QAND\ T=b]\\
    & = \sum_{b \in \range{R}} \pr{R=b \QAND\ S=b \QAND\ T=b}
          & \text{(disjoint sum rule)}\\
    & = \sum_{b \in \range{R}} \pr{R=b}\cdot \pr{S=b} \cdot \pr{T=b}
          & \text{mutual independence}\\
    & = \sum_{b \in \range{R}} p \cdot p  \cdot \pr{T=b}
          & \text{$R,S$ are  uniform}\\
    & = p^2 \sum_{b \in \range{R}} \pr{T=b}\\
    & = p^2 = \pr{R=S} \cdot \pr{R=T}.
\end{align*}
\end{proof}
\end{solution}

\eparts
\end{problem}

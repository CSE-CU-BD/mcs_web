\begin{problem}
How many 4-digit numbers with sum of digits equal to 22 are there?

\hint{Exclusion-Inclusion.}

\examspace[3in]

\begin{solution}
Let $x_1$, $x_2$, $x_3$ and $x_4$ be the digits of a 4-digit number $a$. Let
\begin{align*}
A & = \set{(x_1, x_2, x_3, x_4) \in \integers^4 \,|\, 0 \leq x_1, x_2, x_3, x_4 \leq 9 \QAND x_1+x_2+x_3+x_4 = 22} \\
S & = \set{(x_1, x_2, x_3, x_4) \in \integers^4 \,|\, x_1, x_2, x_3, x_4 \geq 0 \QAND x_1+x_2+x_3+x_4 = 22} \\
S_i & = \set{(x_1, x_2, x_3, x_4) \in \integers^4 \,|\, x_i \geq 10 \QAND x_j \geq 0, j \neq i \QAND x_1+x_2+x_3+x_4 = 22} \\
S_{i,j} & = \set{(x_1, x_2, x_3, x_4) \in \integers^4 \,|\, x_i, x_j \geq 10 \QAND x_k \geq 0, k \neq i,j \QAND x_1+x_2+x_3+x_4 = 22} \, .
\end{align*}
Then, $A = S \setminus (\union_i S_i)$ and
\begin{align*}
|A| & = |S| - \sum_{i=1}^4 |S_i| + \sum_{1\leq i < j \leq 4} |S_{i,j}| \\
     & = {22+4-1 \choose 4-1} - {4 \choose 1} {12+4-1 \choose 4-1} + {4 \choose 2} {2+4-1 \choose 4-1} \\
     & = {25 \choose 3} - 4 {15 \choose 3} + 6 {5 \choose 3} \\
     & = 2300 - 4 \cdot 455 + 6 \cdot 10 = 540 \, .
\end{align*}

\end{solution}

\end{problem}

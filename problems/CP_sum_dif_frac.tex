\begin{problem}
Find a closed-form for the following sums:

\begin{enumerate}

\item $\sum{i=1}^{n} \{ \frac{1}{i+2012} - \frac{1}{i+2013} \}$

\item $\sum{i=1}^{n} i^3$

\end{enumerate}

\begin{solution}

$$ sum{i=1}^{n} \{ \frac{1}{i+2012} - \frac{1}{i+2013} \} =$$

$$ = \frac{1}{2013} - \frac{1}{2014} + frac{1}{2014} - \ldots - \frac{1}{n+2013} $$

$$ = \frac{1}{2013} - \frac{1}{n+2013} $$

\end{solution}

\end{problem}

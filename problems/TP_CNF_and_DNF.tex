\begin{problem}
  \bparts

  \ppart
  The proposition
  \begin{equation*}
    P\eqdef (A\QAND B \QAND \bar{C} \QAND D \QAND \bar{E}) \QOR (\bar{A} \QAND B \QAND \bar{C} \QAND \bar{E})
  \end{equation*}
  is in Disjunctive Normal Form (with two ``$\QAND$-of-literal'' clauses). Find the \textbf{Full}~Disjunctive Normal Form that is equivalent to $P$, and explain your reasoning.
  \hint You shouldn't need a truth table.
  
  \begin{solution}
    What prevents $P$ from being a Full DNF is the second clause not mentioning variable $D$, so let's put in all possibilities for $D$. Using our equivalence axioms, $P$'s second clause can be equivalently written as
    \begin{align*}
      \quad &(\bar{A} \QAND B \QAND \bar{C} \QAND \bar{E}) \leftrightarrow (\bar{A} \QAND B \QAND \bar{C} \QAND \bar{E}) \QAND \True \\
            &\quad \leftrightarrow (\bar{A} \QAND B \QAND \bar{C} \QAND \bar{E}) \QAND (D \QOR \bar{D})\\
            &\quad \leftrightarrow (\bar{A} \QAND B \QAND \bar{C} \QAND \mathbf{D} \QAND \bar{E}) \QOR (\bar{A} \QAND B \QAND \bar{C} \QAND \mathbf{\bar{D}} \QAND \bar{E}),        
    \end{align*}
    where the last line uses the distributive rule. So the Full DNF is
    \begin{align*}
      P &\leftrightarrow (A\QAND B \QAND \bar{C} \QAND D \QAND \bar{E}) \QOR {}\\
        &\mathbin{\phantom{\leftrightarrow}} (\bar{A} \QAND B \QAND \bar{C} \QAND D \QAND \bar{E}) \QOR {}\\
        &\mathbin{\phantom{\leftrightarrow}} (\bar{A} \QAND B \QAND \bar{C} \QAND \bar{D} \QAND \bar{E}).
    \end{align*}
    (Note that writing ``$P = \cdots$'' instead of ``$P \leftrightarrow \cdots$'' is not quite correct: $P$ and the Full DNF are \emph{different} propositions, even though they are logically equivalent.)
  \end{solution}

  \ppart
  How many ``$\QOR$-of-literal'' clauses are there in the \textbf{Full}~\textbf{Con}junctive Normal Form that is equivalent to $P$? Briefly explain your answer. (Please \textbf{don't} write out this Full CNF.)

  \begin{solution}
    Answer: \textbf{$29$ clauses}.
    
    As $P$ has $5$ variables, $P$'s truth table has $2^5 = 32$ rows. We've shown that precisely $3$ rows evaluate to True, since these correspond to the clauses of $P$'s Full DNF. The remaining $2^5 - 3 = 29$ rows correspond to the clauses of $P$'s Full CNF.
  \end{solution}

  \eparts

\end{problem}

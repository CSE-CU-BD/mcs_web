\begin{problem}
Find an example of sets $A$ and $B$, such that $\naturals \strict A \strict B$.
\inhandout{($X \strict Y$ means there is no surjective
function from $X$ to $Y$.)}

\begin{solution}
Let $A$ be any uncountable set and $B$ be the power set of $A$, for
example, $A$ is the real numbers and $B \eqdef \power(\reals)$.
\end{solution}

\end{problem}

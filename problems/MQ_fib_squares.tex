\begin{problem} 

The $n$th Fibonacci number, $F(n)$, is defined recursively as follows:
\begin{align*}
F(0) & \eqdef 0,\\
F(1) & \eqdef 1,\\
F(n+2) & \eqdef F(n+1) + F(n)  & \text{ for } n \ge 0.
\end{align*}
These numbers provide lots of fun for math hobbyists because they
satisfy many unexpected identities, such as
\begin{equation}\label{f02f12fnfn1}
F(0)^2 + F(1)^2 + \cdots + F(n)^2 = F(n) F(n+1).
\end{equation}

\iffalse
Example: $n = 4$:
	\[
	0^2 + 1^2 + 1^2 + 2^2 + 3^2 = 15 = 3 \cdot 5.
	\]
	Let's try a proof by induction.
\fi

Equation~\eqref{f02f12fnfn1} can be proved to hold for all $n \in
\naturals$ by induction, with the equation itself as the induction
hypothesis.

\bparts
\iffalse

\ppart Please give a sensible predicate $P(n)$ for the induction.

	\begin{solution}
	\[
	\sum_{i=0}^n F_i^2 = F_n F_{n+1}.
	\]
	\end{solution}
\fi

\ppart  Prove the \textbf{base case} $(n=0)$.

	\begin{solution}

	$\Sigma_{i=0}^0 F_i^2 \eqdef (F_0)^2 = 0 = F_0 F_1$ because
	$F_0 \eqdef 0$.

	\end{solution}

  \ppart Now prove the \textbf{inductive step}.

\iffalse
Now state $P(n+1)$, and then prove it assuming $P(n)$, thereby proving
the induction step.
\fi

	\begin{solution}
	$P(n+1) = (\Sigma_{i=0}^{n+1}F_i^2 = F_{n+1}F_{n+2}$) \\
	\begin{proof}
	$\Sigma_{i=0}^{n+1}F_i^2 = \Sigma_{i=0}^nF_i^2 + F_{n+1}^2$ \\
	$ = F_nF_{n+1} + F_{n+1}^2$ by $P(n)$ \\
	$ = F_{n+1}(F_n+F_{n+1})$ \\
	$= F_{n+1}F_{n+2}$ by def. of Fibonacci seq.
	\end{proof}

	\end{solution}

\eparts

\end{problem}

\begin{problem}
The set equation 
\[
\overline{A \intersect B} = \overline{A} \union\overline{B}
\]
follows from a certain equivalence propositional between propositional formulas.

\bparts

\ppart What is the equivalence?

\begin{solution}
DeMorgn's Law
\[
\QNOT(A \QAND B) = \QNOT(A) \QOR \QNOT(B).
\]
\end{solution}

\examspace[0.5in]

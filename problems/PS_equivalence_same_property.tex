\begin{problem}
For any total function $f:A \to B$ define a relation $\equiv_f$ by the rule:
\begin{equation}\label{equivf}
a \equiv_f a' \qiff f(a) = f(a').
\end{equation}

\bparts

 \ppart Prove that $\equiv_f$ is an equivalence relation on $A$.

\begin{solution}
Reflexivity, symmetry, and transitivity of $\equiv_f$ follow
immediately from the reflexivity, symmetry, and transitivity of the
equality relation.
\end{solution}

\ppart Prove that every equivalence relation, $R$, on a set, $A$, is
equal to $\equiv_f$ for some total function $f:A \to B$.

\hint Let $f(a) \eqdef [a]_R$ where $[a]_R$ is the equivalence
class\footnote{See Definition~\bref{def:equiv_class}.} of $R$ that
contains $a$.

\begin{solution}

\begin{align*}
a \mrel{R} b
   & \qiff [a]_R = [b]_R
           & \text{(by def of equivalence class)}\\
   & \qiff f(a) = f(b)
           & \text{(by def $f$)}\\
   & \qiff a \equiv_f b
           & \text{(by def $\equiv_f$)}.
\end{align*}
That is, $R = \equiv_f$.
\end{solution}

\eparts

\end{problem}

\begin{problem}
  Prove that for all strings $r,s,t \in \strings{A}$
\[
(r \cdot s) \cdot t = r \cdot (s \cdot t).
\]

\begin{solution}
  By structural induction on the definition of $s \in \strings{A}$.  The
  induction hypothesis is
\[
P(r) \eqdef \forall s,t \in \strings{A}.\, (r \cdot s) \cdot t = r \cdot (s \cdot t).
\]

\end{solution}

\end{problem}

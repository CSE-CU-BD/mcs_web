\begin{problem}
In this problem we'll prove that for all pairs of integers $ a,m$ (even if they aren't relatively prime):
\[ a^{m} \equiv a^{m - \phi(m )} \pmod{m}\]
\bparts
\ppart Show that if $ (a,p_i)=1$ then $ p_i^{k_i}|(a^{\varphi(m)}-1)$.
\begin{solution}
This follows from Euler's theorem:

\end{solution}
\ppart Show that if $ p_i|a$ then $ p_i^{k_i}|a^{m-\varphi(m)}.$
\begin{solution}
Let $ A\equal{}a^m-a^{m-\varphi(m)}=a^{m-\varphi(m)}(a^{\varphi(m)}-1)$. Assume $ m=p_1^{k_1}\cdots p_n^{k_n}$. We only need to prove that if $ (a,p_i)=1$ then $ p_i^{k_i}|(a^{\varphi(m)}-1)$, and if $ p_i|a$ then $ p_i^{k_i}|a^{m-\varphi(m)}.$ The first assertion follows from Euler's theorem, while the second is easy to check, since $ m-\varphi(m)\geqslant k_i$ (there are at least $ k_i$ numbers that are not coprime with $ m$: $ p_i,p_i^2,\ldots,p_i^{k_i}$).
\end{solution}

\end{problem}

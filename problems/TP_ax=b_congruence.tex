\begin{problem}
Show that the equation
\[
ax \equiv b \pmod n
\]
is solvable iff $\gcd(a,n) \divides b$

\begin{solution}
\begin{align*}
\exists x.\ ax \equiv b \pmod n
  & \QIFF\ \exists x.\, \exists y. ax + ny = b\\
  & \QIFF\ \exists x.\, \exists y. \gcd(a,n) \divides b
      & \text{(by Corollary~\bref{cor:lin-comb})}.
\end{align*}

\end{solution}
\end{problem}

\begin{problem}
Show that the equation
\[
ax \equiv b \pmod n
\]
is solvable iff $\gcd(a,n) \divides b$.

\begin{solution}
\begin{align*}
\exists x.\ ax \equiv b \pmod n
  & \QIFF\ \exists x.\, \exists y. ax + ny = b\\
  & \QIFF\ \gcd(a,n) \divides b
      \inbook{& \text{(by Corollary~\bref{cor:lin-comb})}}.
\end{align*}

\end{solution}
\end{problem}

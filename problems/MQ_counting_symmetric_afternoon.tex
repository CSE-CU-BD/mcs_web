\begin{problem}

\bparts

\ppart How many digraphs are there with vertex set $\set{1,2,\dots,n}$?\hfill \examrule[0.7in]

Briefly explain.

\examspace[2in]

\begin{solution}
\[
2^{n^2}
\]

There are $n^2$ potential edges, each of which may or
may not appear in a given graph.

\end{solution}

\ppart How many reflexive, symmetric binary relations on the set
$\set{1,2,\dots,n}$ are there?\hfill \examrule[0.7in]

Briefly explain.

\begin{solution}
\[
2^{\binom{n}{2}}.
\]

All elements are related to themselves in a reflexive relation.  There
are $\binom{n}{2}$ sets of two integers which may or may not be
related.
\end{solution}

\eparts
\end{problem}

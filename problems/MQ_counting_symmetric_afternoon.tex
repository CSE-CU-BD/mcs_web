\begin{problem}
This problem is about binary relations on the set of integers in the
interval $[1,n]$, and digraphs and simple graphs whose vertex set is
$[1,n]$.

\bparts

\ppart How many digraphs are there?

\begin{solution}
There are $n^2$ potential edges, each of which may or
may not appear in a given graph.  Therefore, the number of graphs is:
\[
2^{n^2}
\]
\end{solution}

\ppart How many symmetric binary relations are there?

\begin{solution}
There are $\binom{n}{2}$ pairs of distinct elements $a$ and $b$ which
may or may not be related.  Therefore, the number of asymmetric binary relations is
\[
2^{\binom{n}{2}}.
\]
\end{solution}

\eparts
\end{problem}

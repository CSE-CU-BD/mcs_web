\begin{problem}

Prove by induction:
\end{problem}

\begin{equation}
	\sum_{i=1}^n (3i-1)^2 = \dfrac{n(6n^2+3n-1)}{2}, \forall n \geq 1.
\end{equation}
using the equation itself as the induction hypothesis, $P(n)$. 

\bparts

\ppart Prove the \textbf{base case} $(n=0)$.

\begin{solution}
\[
\sum_{i=1}^1 (3i-1)^2 = 4 = \dfrac{1\cdot 8}{2}.
\]
Therefore, $P(1)$ is true.
\end{solution}

\examspace[2in]

\ppart Now prove the \textbf{inductive step}.

\begin{solution}
\begin{align*}
\sum_{i=1}^{n+1} (3i-1)^2 
& = \sum_{i=1}^{n} (3i-1)^2 + (3n+2)^2 
& = \frac{(6n^3+3n^2-n)}{2} + (3n+2)^2 \text{ (by induction hypothesis)}\\
& = \frac{(6n^3+3n^2-n)}{2} + (9n^2+12n+4) = \frac{(6n^3+21n^2+23n+8)}{2}\\
& = \frac{(n+1)(6n^2+15n+8)}{2} = \frac{(n+1)(6(n+1)^2+3(n+1)-1)}{2}.
\end{align*}
Thus we have proved $P(n+1)$, completing the induction step. 
We conclude by induction that $P(n)$ holds for all $n \in \naturals^{+}$.
\end{solution}

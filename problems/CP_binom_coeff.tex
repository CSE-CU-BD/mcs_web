\begin{problem}
Find the coefficients of 

\bparts

\ppart
 $x^5$ in $(1+x)^{11}$

\begin{solution}
\[
\binom{11}{5} = 462
\]
\end{solution}

\ppart $x^8y^9$ in $(3x+2y)^{17}$

\begin{solution}
\[
\binom{17}{8}3^82^9
\]
\end{solution}

\ppart $a^6b^6$ in $(a^2+b^3)^5$
\begin{solution}
$a^6b^6 = (a^2)^3 (b^3)^2$, so the coefficient is
\[
\binom{5}{3} =10
\]
\end{solution}

\eparts
\end{problem}

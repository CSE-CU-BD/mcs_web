\begin{problem}
Consider a stable marriage problem with 4 boys and 4 girls. Here are their
preference rankings:

\begin{center}
\begin{tabular}{r|c} 
Alfred:    & Grace, Helen, Emily, Fiona    \\
Billy:     & Emily, Grace, Fiona, Helen    \\
Calvin:    & Helen, Emily, Fiona, Grace    \\
David:     & Helen, Grace, Emily, Fiona
\end{tabular}
\end{center}

\begin{center}
\begin{tabular}{r|c} 
Emily:     & Calvin, Alfred, David,  Billy    \\
Fiona:     & Alfred, Billy,  Calvin, David    \\
Grace:     & Alfred, Calvin, David,  Billy    \\
Helen:     & Alfred, Billy,  David,  Calvin
\end{tabular}
\end{center}

\bparts

\ppart
Find \emph{one} stable matching between the boys and girls using the Mating Ritual.
\vspace{2in}

\begin{solution}
  The Mating Ritual reveals that the boy-optimal and boy-pessimal 
  matchings are the same:
  \center (Alfred, Grace), (Billy, Fiona), (Calvin, Emily), (David, Helen)
\end{solution}


\ppart
Find a \emph{different} stable matching, or explain why there is none.
\vspace{2in}

\begin{solution}
  The boy-optimal and boy-pessimal assignments are the same, so the matching
  in the previous part is \emph{unique}.
\end{solution}

\eparts

\end{problem}

\begin{problem}
Prove that for finite random walk graphs, the uniform distribution is stationary if and only
the probabilities of the edges coming into each vertex always sum to 1, namely
\begin{equation}\label{eq:sumuvin}
 \sum_{u \in \vin{v}} p(u,v)= 1,
\end{equation}
where $\vin{w} \eqdef \set{v \suchthat \diredge{v}{w} \text{ is an edge}}$.

\begin{solution}
Suppose $d$ is a uniform distribution, that is, $d(u) = d(v)$ for all
vertices $u,v$.  Then
\begin{align*}
\widehat{d}(v)
 & \eqdef \sum_{u \in \vin{v}} p(u,v)d(u)\\
 & = \sum_{u \in \vin{v}} p(u,v)d(v)\\
 & = d(v)\sum_{u \in \vin{v}} p(u,v),
\end{align*}
and so $d$ is stationary iff $\widehat{d}(v) = d(v)$
iff~\eqref{eq:sumuvin} holds.
\end{solution}

\end{problem}

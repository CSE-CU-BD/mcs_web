\begin{problem} 
Let $R$ be a weak partial relation on a set, $A$.  Suppose $C$ is a
finite chain.\footnote{Elements $a$ and $b$ are \emph{comparable} with
  respect to $R$ iff $[a \mrel{R} b\ \QOR\ b \mrel{R} a]$.  A set $C$
  is a \emph{chain} when all elements $c_0,c_1 \in C$ are comparable.}

\bparts
\ppart Prove that $C$ has a maximum element.

\ppart Conclude that there is a unique sequence of all the elements of
$C$ that is strictly increasing.

\iffalse A partial
  order for which every two different elements are comparable is
  called a \emph{linear order}\fi

\eparts

\end{problem}

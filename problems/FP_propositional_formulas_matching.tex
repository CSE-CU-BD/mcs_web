\begin{problem}
Let $R$ be the ``implies'' binary relation defined by the rule
\begin{equation}
F \mrel{R} G \qiff [\text{the formula } (F \QIMPLIES G) \text{ is valid}].
\label{eq:rule_implies}
\end{equation}
For example, $(P \QAND Q) \mrel{R} P$, because the formula $(P \QAND
Q)$ does imply $P$.  Also, it is not true that $(P \QOR Q) \mrel{R} P$
since $(P \QOR Q)$ does not imply $P$.

\bparts

\ppart\label{ppart:fig} Fill in the arrows so that the following
figure describes the graph of the relation $R$ from $A$ to $B$.

\[\begin{array}{lcr}
A & \hspace{1in} \text{arrows} \hspace{1in} & B\\
\hline
% &\\
% M\\
% &\\
                                  && Q\\
&\\
P \QXOR Q\\
&\\
                                  && \bar{P} \QOR \bar{Q}\\
&\\
P \QAND Q\\
&\\
                                  && \bar{P} \QOR \bar{Q} \QOR
                                  (\bar{P} \QAND \bar{Q})\\
&\\
\QNOT(P \QAND Q)\\
&\\
                                  && P\\
\end{array}\]

\begin{solution}
Arrows for $R$:
\begin{align*}
P \QXOR Q & \qimplies & \bar{P} \QOR \bar{Q}\\
P \QXOR Q & \qimplies &  \bar{P} \QOR \bar{Q} \QOR (\bar{P} \QAND
                                  \bar{Q})\\
P \QAND Q & \qimplies & Q\\
P \QAND Q & \qimplies & P\\
\QNOT(P \QAND Q) & \qiff & \bar{P} \QOR \bar{Q} \\
\QNOT(P \QAND Q) & \qiff &  \bar{P} \QOR \bar{Q} \QOR (\bar{P} \QAND
                                  \bar{Q})
\end{align*}
\end{solution}

\ppart Let $R'$ be the binary relation on the set $A \union B$ defined
by the rule~\eqref{eq:rule_implies}.  Explain why $R'$ is not a
partial order.

\examspace[1in]

\begin{solution}
$R'$ is not a partial order since the condition of antisymmetry is violated; for
example, $\QNOT(P \QAND Q) \qiff \bar{P} \QOR \bar{Q}$ but the two
formulas are not equal. 
\end{solution}

\ppart We can think of the figure in part~\eqref{ppart:fig} as a
bipartite graph $G$ with the formulas in $A$ constituting the left
vertices, the formulas in $B$ constituting the right vertices, and
the arrows denote the edges. Exhibit a subset $S \in A$ of the left
vertices whose image is the same size as $S$, i.e., $|S| = |N(S)|$. 

\examspace[1.5in]
\begin{solution}
$S = \set{P \QXOR Q, \QNOT(P \QAND Q)}$,  $N(S) = \set{\bar{P} \QOR \bar{Q}, \bar{P} \QOR \bar{Q} \QOR
  (\bar{P} \QAND \bar{Q})}$. Therefore, $|S| = |N(S)| = 2$.
\end{solution}

\ppart Choose the option that correctly completes the following
statement: Let $G$ be a bipartite graph and let $S$ be a subset of $L(G)$ such
that $|S| = |N(S)|$. Then, a matching in $G$ can exist only if \underline{\hspace{1cm}}.
\begin{romanlist}
\item $|S| > |N(S)|$
\item $|\bar{S}| > \bar{N(S)}$.
\item $|\bar{S}| \leq |N(\bar{S})|$.
\item $|\bar{S}| \leq |\bar{N(S)}| \intersect |N(\bar{S})|$
\end{romanlist}

\begin{solution}
$|\bar{S}| \leq |\bar{N(S)}| \intersect N(\bar{S})$.
\end{solution}

\eparts

\end{problem}

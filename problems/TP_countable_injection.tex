\begin{problem}
Prove that if $S \inj \naturals$\iffalse there is a total injective
($[\geq 1 \text{ out}, \leq 1 \text{ in}]$) relation from $S \to
\naturals$\fi , then $S$ is countable.

\begin{solution}
We know that $S \inj \naturals$ iff $\naturals \surj S$ \inbook{by
  Lemma~\bref{surjinjbij_properties}}, which implies that $S$ is
countable\inbook{ by Lemma~\bref{NsurjC}}.
\end{solution}

\end{problem}

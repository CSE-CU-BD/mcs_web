\begin{problem} Prove by induction:
\begin{equation}\label{1n2}
1+\frac{1}{4}+\frac{1}{9}+\cdots+\frac{1}{n^2}  <  2-\frac{1}{n},
\end{equation}
for all $n >1$.


\begin{solution}
\begin{proof}
(By Induction).  The induction hypothesis is $P(n)$ is the
inequality~\eqref{1n2}.

\textbf{Base Case}: ($n = 2$).  The LHS of\eqref{1n2} in this case is
$1 + 1/4$ and the RHS is $2 - 1/2$.  Since LHS $= 5/4 < 6/4 = 3/2 =$ RHS, 
inequality~\eqref{1n2} holds, and $P(2)$ is proved.

\textbf{Inductive Step}:  Let $n$ be any natural number greater than $1$, and
assume $P(n)$ in order to prove $P(n+1)$.  That is, we assume~\eqref{1n2}
Adding $1/(n+1)^2$ to both sides of this inequality yields

\begin{align*}
1+\frac{1}{4}+\cdots+\frac{1}{n^2}+\frac{1}{(n+1)^2}
& < 2-\frac{1}{n} +\frac{1}{(n+1)^2}\\
& = 2 - \paren{\frac{1}{n} - \frac{1}{(n+1)^2}}\\
& = 2 - \paren{\frac{n^2 + 2n + 1-n}{n(n+1)^2}}\\
& = 2 - \frac{n^2 + n}{n(n+1)^2}-\frac{1}{n(n+1)^2}\\
& = 2 - \frac{1}{n+1} - \frac{1}{n(n+1)^2}\\
& < 2 - \frac{1}{n+1}.
\end{align*}
So we have proved $P(n+1)$.
\end{proof}

\end{solution}
\end{problem}

\begin{problem}

%% type: short-answer
%% title: Power Sets

For any set $A$, let $\mathcal{P}(A)$ be its \emph{power
set}, the set of all its subsets; note that $A$ is itself a member of
$\mathcal{P}(A)$. Let $\emptyset$ denote the empty set.

\bparts

\ppart
The elements of $\mathcal{P}(\{1,2\})$ are:

\begin{solution}
$\{1,2\}$, $\{1\}$, $\{2\}$, $\emptyset$
\end{solution}

\ppart
The elements of $\mathcal{P}(\{\emptyset, \{\emptyset\} \})$ are:

\begin{solution}
$\{\emptyset, \{\emptyset\} \}$,
$\{\emptyset\}$,
$\{ \{\emptyset\} \}$,
$\emptyset$
\end{solution}

\ppart
How many elements are there in $\mathcal{P}(\{1, 2, \dots , 8\})$?

(You may answer with an integer or an infix expression like
``\verb+5 * 7^3+'')

\begin{solution}
$2^8$
\end{solution}

\eparts

\end{problem}

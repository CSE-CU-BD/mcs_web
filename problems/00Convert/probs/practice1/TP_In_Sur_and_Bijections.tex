\begin{problem}

%% type: short-answer
%% title: In- ,Sur-, and Bijections

For each of the following real-valued functions on the
real numbers, indicate whether it is a \textbf{B}ijection, a
\textbf{S}urjection but not a bijection, an \textbf{I}njection but not a
bijection, or \textbf{N}either an injection nor a surjection.

\bparts

\ppart
$x+2$

\begin{solution}
B
\end{solution}

\ppart
$2x$

\begin{solution}
B
\end{solution}

\ppart
$x^2$

\begin{solution}
N

not a surjection, since negative numbers are not squares of reals, and
not an injection since $(-1)^2=1^2$.
\end{solution}

\ppart
$x^3$

\begin{solution}
B
\end{solution}

\ppart
$\sin x$

\begin{solution}
N

not a surjection, since $-1 \leq \sin x \leq 1$, and not an injection,
since $\sin 0 = \sin 2\pi$.
\end{solution}

\ppart
$x\sin x$

\begin{solution}
S
not an injection, since $0 \sin 0 = 2 \pi \sin 2 \pi$.
\end{solution}

\ppart
$e^x$

\begin{solution}
I

not a surjection, since $e^x$ is always positive for real values of
$x$.
\end{solution}

\eparts

\end{problem}

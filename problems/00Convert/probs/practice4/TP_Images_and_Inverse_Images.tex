\ppart
%% type: short-answer
%% title: Divisibility Images

Let $V$ be the relation with the integers from 7 to~15, and codomain
the integers from 2 to~30, such that $m\mathcal{V}xn$ is true iff $m$
is a divisor of~$n$.

\begin{enumerate}

\item
List the elements of $\{10,14\}\mathcal{V}$, the image of the set
$\{10,14\}$ under $\mathcal{V}$.

%%  Answer with a sequence of integers in any order separated by spaces.
%%  For example, 9 7 11

\begin{solution}

10,
20,
30,
14,
28

\end{solution}

\item
List the elements of $\mathcal{V}{10,14}$, the inverse image of
$\{10,14\}$ under $\mathcal{V}$.

\begin{solution}

7,
10,
14

\end{solution}

\end{enumerate}

(check all that apply):
\begin{enumerate}

\item $\mathcal{R}A=B$

\begin{solution}

false

\end{solution}

\item $\mathcal{R}B = A$

\begin{solution}
true

See Section~4.6 of the lecture notes.
\end{solution}

\item
$A\mathcal{R} = B$

\begin{solution}
false
\end{solution}

\item $B\mathcal{R} = A$

\begin{solution}
false
\end{solution}

\end{enumerate}

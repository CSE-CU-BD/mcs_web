\begin{problem}

%% type: short-answer
%% title: Generating Functions and Sequences

Consider the following generating functions:

\begin{description}

\item[a]
$x^{2}/(1-x)$

\item[b]
$e^{x}$

\item[c]
$(1+x)^{4}$

\item[d]
$(1+x)(1-x)^{-3}$

\item[e]
$1/(1-x)^{2}$

\item[f]
$1+x$

\item[g]
$1/(1-x^{2})$

\end{description}

For each of the following sequences, indicate
the corresponding generating function 
by entering one of the letters \textbf{a b c d e f g}.

\bparts

\ppart
0, 0, 1, 1, 1,\dots

\begin{solution}

a

\end{solution}

\ppart
1, 1, 0, 0, 0,\dots

\begin{solution}

f

\end{solution}

\ppart
1, 0, 1, 0, 1, 0, 1,\dots

\begin{solution}

g

\end{solution}

\ppart
1, 4, 6, 4, 1, 0, 0, 0,\dots

\begin{solution}

c

\end{solution}

\ppart
1, 1, 1/2, 1/6, 1/24, 1/120,\dots

\begin{solution}

b

\end{solution}

\ppart
1, 2, 3, 4, 5,\dots

\begin{solution}

e

Derivative of $1/(1-x)$
\end{solution}

\ppart
1, 4, 9, 16, 25,\dots

\begin{solution}

d

Derivative of $x(1-x)^{-2}$
\end{solution}

\eparts

\end{problem}

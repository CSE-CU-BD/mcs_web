\begin{problem}

%% type: short-answer
%% title: Euler's Theorem

\begin{enumerate}

\item
What is the value of $\phi(175)$, where $\phi$ is Euler's function?

\begin{solution}
120

Noting that $175 = 5^{2} \cdot 7$, it follows that $\phi(175)=(5^{2} -
5^{1})(7-1) = 20 \cdot 6 = 120$.
\end{solution}

\item
What is the remainder of $22^{12001}$ divided by 175?

\begin{solution}

22

\begin{align*}
22^{12001} &= 22^{(120 \cdot 100) +1} \\
           & = (22^{120})^{100}  \cdot  22 \\
           & \equiv  1^{100}  \cdot  22 \\
           & \equiv  22 \bmod 175.
\end{align*}
\end{solution}

\end{enumerate}

\begin{problem}

%% type: short-answer
%% title: Divisors

\begin{enumerate}

\item
How many prime divisors does 12 have?

\begin{solution}
2

The prime divisors are 2 and 3.
\end{solution}

\item
How many positive divisors does 12 have?

\begin{solution}
6

The positive divisors are 1, 2, 3, 4, 6, and 12.
\end{solution}

\item
How many divisors does 12 have?

\begin{solution}

12

The divisors are 
 $\pm 1$,  $\pm 2$,  $\pm 3$,  $\pm 4$,  $\pm 6$, and  $\pm 12$.
\end{solution}

\end{enumerate}

\end{problem}

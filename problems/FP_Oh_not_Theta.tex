\begin{problem} \mbox{}

 \bparts

 \ppart Define a function $f(n)$ such that $f = \Theta(n^2)$
  and $\QNOT (f \sim n^2)$.

\begin{solution}
Let $f(n) \eqdef 2n^2$.
\end{solution}

\iffalse

  \examspace[0.3in]
  \[
  f(n) = \exambox{4.0in}{0.75in}{-0.35in}
         \insolutions{2n^2}
  \]

  \examspace[0.3in]
\fi

 \ppart Define a function $g(n)$ such that $g = O(n^2)$,
  $g \neq \Theta(n^2)$, $g \neq o(n^2)$, and $n = O(g)$.

\begin{solution}
Let $g(n) \eqdef \paren{n\sin(n\pi/2)}^2 + n$.

For odd $n$, we have $g(n) = n^2$, which implies that $g \neq
o(n^2)$.  For even $n$, we have $g(n) = n$, which implies $n^2 \neq O(g)$
and hence $g \neq \Theta(n^2)$.

\end{solution}

\iffalse

  \examspace[0.3in]
  \[
  g(n) = \exambox{4.0in}{0.75in}{-0.35in}
         \insolutions{n^2\sin(\frac{n\pi}{2}) + n\cos(\frac{n\pi}{2})}
  \]
\fi

 \eparts
\end{problem}

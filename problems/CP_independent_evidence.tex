\begin{problem}
Event $E$ is \emph{evidence} for event $H$ when $\prcond{H}{E} >
\pr{H}$.

\bparts

\ppart Give an example of events $A,B,H$ such that $A$ and $B$ are
independent, both are evidence for $H$, but $A \union B$ is not
evidence for $H$.

\hint Let $\sspace = \Zintv{1}{8}$

\begin{staffnotes}
There may be simpler examples.
\end{staffnotes}

\begin{solution}
\begin{align*}
H & \eqdef \set{3,4,8}\\  %\set{3,4}\\
A & \eqdef \set{1,2,3,4}\\
B & \eqdef \set{3,4,5,6}.
\end{align*}
\end{solution}

\ppart Prove that there is no event $E$ such that $E$ and
$\setcomp{E}$ are both evidence for $H$.

\begin{solution}
Let
\[
m \eqdef \min\paren{\prcond{H}{E}, \prcond{H}{\setcomp{E}}}.
\]
Then
\begin{align*}
\pr{H} & = \prcond{H}{E}\pr{E} + \prcond{H}{\setcomp{E}}\pr{\setcomp{E}}
             & \text{(Total probability)}\\
       & \geq m\pr{E} + m\pr{\setcomp{E}}\\
       & = m.
\end{align*}
So,
\[
\pr{H} \geq m.
\]
But if $E$ and $\setcomp{E}$ were both
evidence for $H$, we have
\[
m > \pr{H},
\]
a contradiction.
\end{solution}

\eparts

\end{problem}

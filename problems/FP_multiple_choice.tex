The following statements about trees:

\begin{itemize}
\item Any connected subgraph is a tree.

\textbf{true}

\textbf{false}: \instatements{\vspace{0.4in}}

\item Adding an edge between two vertices creates a cycle.

\textbf{true}

\textbf{false}: \instatements{\vspace{0.4in}}

\item The number of vertices is one less than twice the number of leaves.

\textbf{true}

\textbf{false}: \instatements{\vspace{0.4in}}

\end{itemize}

\begin{solution}``The number of all the vertices is one less than
twice the number of leaf nodes'' is not true.  This property holds for
full binary trees, but not in general.  A tree with two vertices is a
counterexample.
\end{solution}

\begin{itemize}

\item If $\gcd(a, b) \neq 1$ and $\gcd(b, c) \neq 1$, then $\gcd(a, c) \neq 1$.

\textbf{true}

\textbf{false}: \instatements{\vspace{0.4in}}

\textbf{true}

\textbf{false}: \instatements{\vspace{0.4in}}

\item If $a \divides b c$ and $\gcd(a, b) = 1$, then $a \divides c$.

\textbf{true}

\textbf{false}: \instatements{\vspace{0.4in}}

\item $\gcd(a^n,b^n) =  (\gcd(a,b))^n$ 

\textbf{true}

\textbf{false}: \instatements{\vspace{0.4in}}

%%Removed

The following statements about equivalence mod $n$, where $n > 1$.

\begin{itemize}

\item If $a c \equiv b c \pmod{n}$, then $a \equiv b \pmod{n}$.

\textbf{true}

\textbf{false}: \instatements{\vspace{0.3in}}

\item If $a \equiv b \pmod{\phi(n)}$ for $a, b > 0$, then $c^a \equiv c^b
\pmod{n}$.

\textbf{true}

\textbf{false}: \instatements{\vspace{0.3in}}

\item If $a \equiv b \pmod{n}$, then $P(a) \equiv P(b) \pmod{n}$ for any
polynomial $P(x)$ with integer coefficients.

\textbf{true}

\textbf{false}: \instatements{\vspace{0.3in}}

\item If $a \equiv b \pmod{nm}$, then $a \equiv b \pmod{n}$, for $m \geq 1$.

\textbf{true}

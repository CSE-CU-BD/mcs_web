\begin{problem}
Prove that
\[
f = \Theta(g)\ \QIMPLIES\ \ln f \sim \ln g,
\]
for all nonnegative real-valued functions $f,g$.

\begin{solution}
$f = \Theta(g)$ implies that there are constants $a,b$ such that
\[
af(x) \leq g(x) \leq bf(x)
\]
for all large $x \in \reals$.  Assuming $f,g \geq 1$ so $\ln f, \ln g \geq 0$, and taking logs of the expressions  above, we get
\[
\ln a + \ln f(x) \leq \ln g(x) \leq \ln b + \ln f(x)
\]
so
\[
\frac{\ln g}{\ln f}
\]
is bounded above and below by constants, which implies $\ln g \sim \ln
f$.

A similar argument applies in the case that $f,g < 1$.

\end{solution}

\end{problem}

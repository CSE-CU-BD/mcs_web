\begin{problem}

%% type: short-answer
%% title: Binomial Random Variable

Let $B$ be the number of heads that come up on $2n$ independent tosses of a fair coin.
\bparts


\ppart
$\prob{B = n}$ is asymptotically equal to one of the expressions given below. Which one? 
      

      
\begin{enumerate}



      
\item
$\frac{1}{\sqrt{2 \pi n}}$

      

\item
$\frac{2}{\sqrt{\pi n}}$

      

\item
$\frac{1}{\sqrt{\pi n}}$

      

\item
$\sqrt{\frac{2}{\pi n}}$

      
\end{enumerate}



\begin{solution}

3.


We have that $\prob{B = n} = {{2n} \choose {n}} * 2^{-2n}$. Using Stirling's Formula, 
\begin{equation*}
{{2n} \choose {n}} \sim \frac{2^{2n H(1/2)}}{\sqrt{2 \pi (1/2) (1 - 1/2) 2n}} = \frac{2^{2n}}{\sqrt{\pi n}}
\end{equation*}
Thus, 
\begin{equation*}
\prob{B = n} \sim \frac{2^{2n}}{\sqrt{\pi n}} * 2^{-2n} = \frac{1}{\sqrt{\pi n}}.
\end{equation*}
\end{solution}

\eparts


\end{problem}

\begin{problem}

\bparts

\ppart Use the \idx{Pulverizer} to find $gcd(84,108)$

\examspace[3in]

\begin{solution}
Here is the table produced by the Pulverizer:
\[
\begin{array}{ccccrcl}
x & \quad & y & \quad & \rem{x}{y} & = & x - q \cdot y \\ \hline
 108  &&  84  &&  24  & = &  1 \cdot 108 - 1 \cdot 84   \\
  84  &&  24  &&  12  & = & -3 \cdot 108 + 4 \cdot 84   \\
  24  &&  12  &&  0  & & 
\end{array}
\]
\end{solution}

\ppart Find integers $x$, $y$ with $168 \leq y < 252$ such that
\[
x \cdot 84 + y \cdot 108 = \gcd(84,108).
\]

\examspace[2in]

\begin{solution}
From the table above, 
\[
4 \cdot 84 - 3 \cdot 108 = \gcd(84,108).
\]
Therefore,
\[
(4 - 108 \cdot k) \cdot 84 + (-3 + 84 \cdot k) \cdot 108 = \gcd(84,108).
\]
So, letting $k = 3$, $(x,y) = (4 - 108 \cdot 3, -3 + 84 \cdot 3) = (-320, 249)$ works.
\end{solution}

\eparts
\end{problem}

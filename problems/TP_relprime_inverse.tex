\begin{problem}
Prove that if $k\in [0,n)$ has an inverse modulo $n$, then it has an
  inverse in $\Zmod{n}$.

\begin{solution}
Suppose $1 \equiv jk \pmod{n}$.  Then $1 \equiv j'k \pmod{n}$ where
$j' = \rem{j}{n} \in [1,n)$.  then by Lemma~\bref{lem:conrem}, $1 =
  \rem{j'k}{n}$.  But $j' \cdot_n k \eqdef \rem{j'k}{n}$, so $j'$ is
  an inverse of $k$ in $\Zmod{n}$.
\end{solution}

\end{problem}

\begin{problem}
\bparts

Use the equivalence axioms of
Section~\bref{propositional_equivalences_sec} to convert the
formula
\[
A \QXOR B \QXOR C.
\]

\ppart \dots to disjunctive (\QOR\ of \QAND's) form,

\begin{solution}

Noting that $A$ \textsc{XOR} $B$ \textsc{XOR} $C$ evaluates to \textbf{T} exactly when any one of the inputs is \textbf{T} or all three are \textbf{T}: $(A \wedge \overline{B} \wedge \overline{C}) \vee (\overline{A} \wedge B \wedge \overline{C}) \vee (\overline{A} \wedge \overline{B} \wedge C) \vee (A \wedge B \wedge C)$.

\end{solution}

\ppart \dots to conjunctive (\QAND of\ \QOR's) form.

\begin{solution}
Using DeMorgan's theorem, and knowing that $A$ \textsc{XOR} $B$ \textsc{XOR} $C$ evaluates to \textbf{F} exactly when any two of the variables are \textbf{F} or all of the variables are \textbf{F}: $\overline{(A \wedge B \wedge \overline{C})} \wedge \overline{(A \wedge \overline{B} \wedge C)} \wedge \overline{(\overline{A} \wedge B \wedge C)} \wedge \overline{(\overline{A} \wedge \overline{B} \wedge \overline{C})}$ $\Rightarrow$ \fbox{$(\overline{A} \vee \overline{B} \vee C) \wedge (\overline{A} \vee B \vee \overline{C}) \wedge (A \vee \overline{B} \vee \overline{C}) \wedge (\overline{A} \vee \overline{B} \vee \overline{C})$}. 
\end{solution}

\eparts

\end{problem}

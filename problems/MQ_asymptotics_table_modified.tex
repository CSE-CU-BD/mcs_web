\begin{problem}

For each pair of functions $f(n)$ and $g(n)$ in the table below, indicate
which of the indicated \idx{asymptotic relations} hold \textbf{and} which do not.
(Fill \textbf{every} cell in the table.  You may use crosses and checkmarks, T(rue) and
F(alse), or Y(es) and N(o).)

\[
\begin{array}{|c|c|c|c|c|c|}
\hline
f(n) & g(n) & f = O(g) & f = o(g) & g = O(f) & g = o(f) \\
\hline\hline
\log_4{n} & \sqrt[3]{n} & & & & \\ \hline
n^2 + 3^n & n^3 + 2^n & & & & \\ \hline
n\ln{n!} & n^2 \log_{10}{n^2} & & & & \\ \hline
n^{2\cos{(\pi n/2)}+3} & 5n^5+3n^3+n & & & & \\ \hline
\end{array}
\]

\begin{solution}

%[\vspace{3in}]
\[
\begin{array}{|c|c|c|c|c|c|}
\hline
f(n) & g(n) & f = O(g) & f = o(g) & g = O(f) & g = o(f) \\
\hline\hline
\log_4{n} & \sqrt[3]{n} & YES & YES & NO & NO \\ \hline
n^2 + 3^n & n^3 + 2^n & NO & NO & YES & YES \\ \hline
n\ln{n!} & n^2 \log_{10}{n^2} & YES & NO & YES & NO \\ \hline
n^{2\cos{(\pi n/2)}+3} & 5n^5+3n^3+n & YES & NO & NO & NO \\ \hline
\end{array}
\]

\end{solution}

\end{problem}

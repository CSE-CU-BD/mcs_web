\begin{problem}[Deviations from the mean]

\iffalse
\bparts

\ppart Prove that if $X$ is a random variable with mean $\mu$,
then $X-\mu$ is a random variable with mean 0.

\begin{solution}
$E(X-m) = E(X) - E(m) = m - m = 0$. Thus $X-m$ is a random variable
with mean $0$.

\end{solution}

\ppart
\fi

Let $B$ be a random variable with \idx{unbiased binomial distribution}, nemely,
\[
\prob{B = k} = \binom{n}{k}2^{-n}\ .
\]
Assume $n$ is even.  Prove the following formula for the \idx{expected
  absolute deviation} of $B$ from its mean:
\[
\expect{\abs{B - \expect{B}}} =
\binom{n}{\frac{n}{2}}\frac{n}{2^{n+1}}.
\]

\begin{solution}

\begin{proof}
\begin{align*}
\lefteqn{\expect{\abs{B - n/2}}}\\
   & = 2^{-n} \brac{\sum_{k=0}^n \binom{n}{k} \abs{k - \frac{n}{2}}}\\
   & = 2^{-n}\brac{2 \sum_{k=0}^{n/2-1} \binom{n}{k} \paren{\frac{n}{2} - k} }\\
   & = 2^{-n}\brac{n \sum_{k=0}^{n/2-1} \binom{n}{k}} +
       2^{-n}\brac{2 \sum_{k=0}^{n/2-1} \binom{n}{k} k}\\
   & = 2^{-n}\brac{n\paren{2^{n-1} - \frac{1}{2}\binom{n}{n/2}}} +
       2^{-n}\brac{2 \sum_{k=1}^{n/2-1} \binom{n-1}{k-1}\paren{\frac{n}{k}}k}\\
   & = 2^{-n}n\brac{2^{n-1} - \frac{1}{2}\binom{n}{n/2}} +
       2^{-n}\brac{2n \sum_{k=0}^{n/2-2} \binom{n-1}{k}}\\
   & = 2^{-n}\brac{n\paren{2^{n-1} - \frac{1}{2}\binom{n}{n/2}}} +
       2^{-n}\brac{2n\paren{2^{n-2} - \binom{n-1}{n/2 - 1}}}\\
   & = 2^{-n}\brac{n\paren{2^{n-1} - \frac{1}{2} \binom{n}{n/2}} +
       2n\paren{2^{n-2} - \binom{n-1}{n/2 - 1}}}\\%big step; better check
   & = 2^{-n}\binom{n}{n/2}\frac{n}{2}\\
   & = \binom{n}{n/2}\frac{n}{2^{n+1}}.
\end{align*}
\end{proof}
\end{solution}

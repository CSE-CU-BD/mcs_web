\begin{problem}
For any random variable, $R$, with mean, $\mu$, and standard deviation,
$\sigma$, the Chebyshev Bound says that for any real number $x > 0$,
\[
\pr{\abs{R-\mu} \geq x} \leq \paren{\frac{\sigma}{x}}^2.
\]
Show that for any real number, $\mu$, and real numbers $x \geq \sigma >0$,
there is an $R$ for which the Chebyshev Bound is tight, that is,
\begin{equation}\label{t}
\pr{\abs{R} \geq x} = \paren{\frac{\sigma}{x}}^2.
\end{equation}

\hint First assume $\mu = 0$ and let $R$ only take values $0, -x,$ and
$x$.

\begin{solution}
From the hint, we aim to find an $R$ with $\expect{R}=0$ and $\variance{R}
= \sigma^2$ that satisfies equation~\eqref{t}.

Using the further hint that $R$ takes only values $0,-x,x$, we have
\[
0 = \expect{R} = x\pr{R=x} - x\pr{R=-x} = x\paren{\pr{R=x} - \pr{R=-x}}
\]
so
\begin{equation}\label{x-x}
\pr{R=x} = \pr{R=-x},
\end{equation}
since $x>0$.  Also,
\[
\sigma^2  = \expect{R^2}  = x^2\pr{R=-x} + x^2\pr{R=x}  = 2x^2\pr{R=x},
\]
so
\[
\pr{R=x} = \frac{\sigma^2}{2x^2}.
\]
This implies
\[
\pr{R=0} = 1 - 2\pr{R=x} = 1- \paren{\frac{\sigma}{x}}^2,
\]
which completely determines the distribution of $R$.  Moreover,
\[
\pr{\abs{R} \geq x} = \pr{R=-x}+\pr{R=x} = 2\pr{R=x} = \paren{\frac{\sigma}{x}}^2
\]
which confirms~\eqref{t}.

Finally, given $\mu,x,$ and $\sigma$, if we let $R'\eqdef R+\mu$, then $R'$
will be the desired random variable for which the Chebyshev Bound is
tight.
\end{solution}

\end{problem}

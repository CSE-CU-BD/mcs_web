\begin{problem}
The \emph{reversal} of a string is the string written backwards, for
example, $\rev{abcde} = edcba$.

\bparts

\ppart Give a simple recursive definition of $\rev{s}$ based on the
recursive definition~\bref{recstring_def} of $s \in \strings{A}$ and using
the concatenation operation~\bref{concat_def}.

\begin{solution}

\begin{itemize}

\item $\rev{\emptystring} \eqdef \emptystring$.
\item
\[
\rev{\ang{a,s}} \eqdef s \cdot a.
\]

\end{itemize}
\end{solution}


\ppart  Prove that 
\[
\rev{s \cdot t} = \rev{t} \cdot \rev{s},
\]
for all strings $s,t \in \strings{A}$.

\end{solution}

\end{problem}

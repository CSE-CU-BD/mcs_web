\begin{problem}

For any binary relation, $\mathcal{R}: A \rightarrow B$ and subset,
$S \subset A$, let $S\mathcal{R}$ be the image of $S$ under $\mathcal{R}$.
%% , that is,
%% 
%% \begin{equation*}
%% S\mathcal{R} ::= \{ b \in B \mid \text{$s\mathcal{R}b$ for some $s \in S$} \}.
%% \end{equation*}

For example, if $\mathcal{R}$ is the doubling function with domain and
codomain equal to the real numbers, namely
\begin{equation*}
    s\mathcal{R}t\quad\text{iff}\quad t=2s,
\end{equation*}
then $\{0,3,11\}\mathcal{R} = \{0,6,22\}$, for example, and
$\mathbb{Z}\mathcal{R}$ is the set of all even integers (here
$\mathbb{Z}$ stands for the integers).  For any finite set, $S$, we
define $|S|$ be the size (number of elements) of~$S$.

Now assume $\mathcal{R}$ is a \emph{total function} and $A$ is finite.
Replace the $\star$s with one of $\{\leq,=,\geq\}$ to produce the
\emph{strongest} correct version of the following statements:

\bparts

\ppart
$|A\mathcal{R}| \star |B|$

\begin{solution}
$|A|\leq |B|$.

since $f(A) \subset B$
\end{solution}

\ppart
If $\mathcal{R}$ is a surjection, then $|A| \star |B|$

\begin{solution}
$|A|\geq |B|$
\end{solution}

\ppart
If $\mathcal{R}$ is a surjection, then $|A\mathcal{R}| \star |B|$

\begin{solution}
$|A|=|B|$
\end{solution}

\ppart
If $\mathcal{R}$ is an injection, then $|A\mathcal{R}| \star |A|$

\begin{solution}
$|A|=|B|$
\end{solution}

\ppart
If $\mathcal{R}$ is a bijection, then $|A| \star |B|$

\begin{solution}
$|A|=|B|$
\end{solution}

\eparts

\end{problem}

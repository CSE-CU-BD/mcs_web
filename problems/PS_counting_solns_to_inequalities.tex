\begin{problem}[3] \mbox{}
\bparts

\ppart For fixed positive integers $n$ and $k$, how many nonnegative 
integer solutions $x_1,x_2,\ldots,x_k$ are there to the following equation? 
\begin{equation}\label{intsoln}
x_1 + x_2 + \ldots + x_k \leq n
\end{equation}

\begin{solution}Over the nonnegative integers, there is a bijection between the 
solutions to $x_1 + \ldots + x_k \leq n$ and the solutions to
$x_1 + \ldots + x_k + x_{k+1} = n$ where $x_{k+1}$ is the amount by
which $n$ exceeds $x_1 + \ldots + x_k$. 

\[
\binom{n+k}{k}
\]
\end{solution}

\ppart Describe a bijection between solutions to \eqref{intsoln} and 
the length $k$ weakly increasing sequences of nonnegative integers $\leq n$.

\begin{solution}Define $y_i=x_1 + x_2 + \ldots + x_i$.\end{solution}

\eparts
\end{problem}

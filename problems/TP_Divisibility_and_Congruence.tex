%% type: short-answer
%% title: Divisibility and Congruence

A majority of the following statements are equivalent.  List all
statements in this majority.  Assume that $n > 0$ and $a$ and $b$ are
integers.
      
\begin{enumerate}
       
\item
 $a \equiv  b \pmod n$
       
\item
 $a = b$
       
\item
 $\rem{a}{n} = \rem{b}{n}$
       
\item
 $n \divides (a - b)$
       
\item
 $\exists k \in \integers.\, a = b + nk$
       
\item
 $(a - b)$ is a multiple of $n$
       
\item
 $n \divides a\ \QOR\ n \divides b$
      
\end{enumerate}

\bparts

\ppart


\begin{solution}

1,3,4,5,6

\end{solution}

\eparts


\end{problem}

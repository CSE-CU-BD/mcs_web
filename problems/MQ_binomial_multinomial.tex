\begin{problem}
\begin{problemparts}

\problempart
Write the term of $(x + y)^{40}$ which includes $x^3$.
\examspace[3in]

\begin{solution}
\[
\binom{40}{3}x^3y^{37}.
\]

\end{solution}

\problempart
Write the term of $(x+y+z)^{40}$ which includes $x^3y^5$.

\begin{solution}
\[
\binom{40}{3,5,32}x^3y^5z^{32}$
\]

\end{solution}

\end{problemparts}

\end{problem}

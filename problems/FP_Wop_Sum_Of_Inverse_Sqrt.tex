\begin{problem}

Prove by WOP that for any natural number $n \geq 2$,

$$ \left( 1 - \frac{1}{\sqrt{2}} \right) \left( 1 - \frac{1}{\sqrt{3}} \right) \ldots \left( 1 - \frac{1}{\sqrt{n}} \right) < \frac{2}{n^2} $$

\bparts

\ppart Define $P(n)$, the property that holds for every nonnegative integer $n \geq 2$:

\begin{solution}

Let 

$$P(n) ::= \left( 1 - \frac{1}{\sqrt{2}} \right) \left( 1 - \frac{1}{\sqrt{3}} \right) \ldots \left( 1 - \frac{1}{\sqrt{n}} \right) < \frac{2}{n^2}$$

\end{solution}

\ppart Define the set $C$ of counterexamples of $P$ being true:

\begin{solution}

$$C = \{ n \geq 2 \in \mathbb{N} | not(P(n)) \} $$

\end{solution}

\ppart Assuming by contraction that $C$ is nonempty, then by the WOP, there will be a smallest element, $n$, in $C$. Reach a contradiction and conclude.

\begin{solution}

For $n=2$:

$$ \textbf{LHS of } P(2) = 1 - \frac{1}{\sqrt{2}} = \frac{\sqrt{2}-1}{\sqrt{2}}$$
$$ = \frac{1}{2 + \sqrt{2}} < \frac{1}{2} = \frac{2}{2^2} = \textbf{RHS of } P(2) $$

Assume that $C$ is nonempty. By WOP there is a smallest element $k+1$, $k \geq 2 \in \mathbb{N}$ in $C$. Therefore by the definition of $C$ 
$P(k+1)$ does not holds. But because $k+1$ is the smallest element in $C$, then $P(k)$ holds, because it's not in $C$ and $k \geq 2 \in \mathbb{N}$.

Therefore we have that:

$$\textbf{LHS of } P(k+1) =  \left( 1 - \frac{1}{\sqrt{2}} \right) \left( 1 - \frac{1}{\sqrt{3}} \right) \ldots \left( 1 - \frac{1}{\sqrt{k}} \right) \left( 1 - \frac{1}{\sqrt{k+1}} \right) $$

$$ = \textbf{(LHS of } P(k)) . \left( 1 - \frac{1}{\sqrt{k+1}} \right) $$

$$ < \textbf{(RHS of } P(k)) .\left( 1 - \frac{1}{\sqrt{k+1}} \right) $$

$$ = \frac{2}{k^2}  .\left( 1 - \frac{1}{\sqrt{k+1}} \right) $$

$$ =\frac{2(\sqrt{k+1} -1)(\sqrt{k+1} + 1)}{k^2 \sqrt{k+1} (\sqrt{k+1} +1)}$$

$$ =\frac{2(k+1-1)}{k^2 \sqrt{k+1} (\sqrt{k+1} +1) }$$

$$ =\frac{2}{k (k+1 +\sqrt{k+1}) }$$

$$ =\frac{2}{k (k+1) +k\sqrt{k+1} }$$

$$ \leq \frac{2}{k (k+1) +\sqrt{k+1} \sqrt{k+1} } \textrm{ (for }k \geq 2)$$

$$ = \frac{2}{k (k+1) + (k+1) }$$

$$ = \frac{2}{ (k+1)^2 }$$

$$ = \textbf{(RHS of } P(k+1))$$

Contradiction. Therefore $C$ is empty and $P(n)$ holds for all $n \geq 2 \in \mathbb{N}$.

\end{solution}

\eparts

\end{problem}

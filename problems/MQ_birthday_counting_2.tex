\begin{problem}
Three people are born on three distinct days, say $(d_1,d_2,d_3)$. Assume 365 days.
\bparts

\ppart How many possible $(d_1,d_2,d_3)$ pairs are there?

\begin{solution}
$365*364*363$
\end{solution}

\ppart Now suppose the birthdays are not necessarily  distinct. How many possible $(d_1,d_2,d_3)$ pairs are there?

\begin{solution}
$365^3$
\end{solution}

\eparts

\end{problem}

\begin{problem}

In each case, say whether or not $R$ is a partial order on $A$.  If
so, is it a path-total order?

\begin{problemparts}

\problempart $A=\set{a,b,c}, R=\set{(a,a),(b,a),(b,b),(b,c),(c,c)}.$

\begin{solution}
$R$ is reflexive, transitive, and antisymmetric.  Therefore, $R$ is
a partial order on $A$.  However, it is not a path-total order on $A$ since
neither the pair $(a,c)$ nor $(c,a)$ is in $R$.
\end{solution}

\problempart $A=$\set{the set of all words in English}.
\[
R = \set{(x,y) \in A \times A \text{ and $x$ comes before $y$ alphabetically}}.
\]

\begin{solution}
$R$ is not reflexive (although it is transitive and antisymmetric),
  and thus it is not a partial order on $A$.
\end{solution}


\problempart $A=\reals, R=\set{(x,y) \in \reals \times \reals \QAND
  \abs{x} < \abs{y} \QOR x=y}$.

\begin{solution}
$R$ is reflexive, transitive, and antisymmetric.  Therefore, $R$ is
a partial order on $A$. However, $R$ is not a path-total order on $A$. As a
counterexample, if x = 5, y = -5, then none of the following holds: $x
= y, |x| < |y|, |y| < |x|$.
\end{solution}


\problempart $A=\reals, R=\set{(x,y) \in \reals \times \reals \QAND
  \abs{x} \leq \abs{y}}$.

\begin{solution}
$R$ is reflexive and transitive.  However, it is not antisymmetric.
  As a counterexample, $\abs{-3} \leq \abs{3}, \abs{3} \leq \abs{-3}$,
  but $3 \neq -3$.
\end{solution}

\end{problemparts}

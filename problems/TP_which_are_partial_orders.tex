\begin{problem}

In each of the following examples, $R$ is a binary relation on a set
$A$.  Indicate which examples are partial orders, and when they are,
indicate whether they are linear.

\begin{problemparts}

\problempart $A=\set{a,b,c},\  R=\set{(a,a),(b,a),(b,b),(b,c),(c,c)}.$

\begin{solution}
$R$ is reflexive, transitive, and antisymmetric.  Therefore, $R$ is
a partial order on $A$.  However, it is not a linear order on $A$ since
neither the pair $(a,c)$ nor $(c,a)$ is in $R$.
\end{solution}

\problempart $A=$ the set of all English words,
\[
R = \set{(x,y) \in A \times A \suchthat x \text{ comes before $y$ alphabetically}}.
\]

\begin{solution}
$R$ is not reflexive (although it is transitive and antisymmetric),
  and thus it is not a partial order on $A$.
\end{solution}


\problempart $A=\reals,\  R= \set{(x,y) \in \reals \times \reals \suchthat
  \abs{x} < \abs{y} \QOR x=y}$.

\begin{solution}
$R$ is reflexive, transitive, and antisymmetric.  Therefore, $R$ is a
  partial order on $A$.  However, $R$ is not a linear order on $A$.
  For example, if $x = 5, y = -5$, then none of the following holds:
  $x = y, \abs{x} < \abs{y}, \abs{y} < \abs{x}$.
\end{solution}


\problempart $A=\reals,\ R=\set{(x,y) \in \reals \times \reals \suchthat
  \abs{x} \leq \abs{y}}$.

\begin{solution}
$R$ is reflexive and transitive.  However, it is not antisymmetric.
  As a counterexample, $\abs{-3} \leq \abs{3}, \abs{3} \leq \abs{-3}$,
  but $3 \neq -3$.
\end{solution}

\end{problemparts}

\end{problem}

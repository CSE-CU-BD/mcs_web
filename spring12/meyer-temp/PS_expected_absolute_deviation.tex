\begin{problem}Deviations from the mean:

\bparts

\ppart Prove that if $X$ is a random variable with mean $m$,
then $X-m$ is a random variable with mean 0.

\begin{solution}
$E(X-m) = E(X) - E(m) = m - m = 0$. Thus $X-m$ is a random variable
with mean $0$.

\end{solution}

\ppart Suppose I flip $n$ unbiased independent coins.  We know that
the mean number of heads is $n/2$.  What is the {\em expected absolute
  deviation from the mean}?  That is, if the number of heads is a
random variable $X$, what is
\[
\expect{\abs{\frac{X-n}{2}}}\ ?
\]

\begin{solution}
Assume $n$ is even.

\begin{eqnarray*}
E(|X - n/2|) & = 2^{-n}\left[ {\displaystyle \sum_{k=0}^n {n\choose k}
    |k - \frac{n}{2}|} \right] & \\
& = 2^{-n}\left[2 {\displaystyle
    \sum_{k=0}^{\frac{n}{2}-1} {n\choose k} (\frac{n}{2} - k) }
  \right] & \\
& = 2^{-n}\left[n{\displaystyle
    \sum_{k=0}^{\frac{n}{2}-1} {n\choose k} }\right] +
2^{-n}\left[2{\displaystyle \sum_{k=0}^{\frac{n}{2}-1} {n\choose k}k}
  \right] & \\
& = 2^{-n}\left[n(2^{n-1} - (\frac{1}{2}){n\choose
    \frac{n}{2}}) \right] + 2^{-n}\left[2{\displaystyle
    \sum_{k=1}^{\frac{n}{2}-1} {n-1\choose k-1}(\frac{n}{k})k} \right] & \\
& = 2^{-n}\left[n(2^{n-1} - (\frac{1}{2}){n\choose \frac{n}{2}})
  \right] + 2^{-n}\left[2n{\displaystyle \sum_{k=0}^{\frac{n}{2}-2}
    {n-1\choose k}} \right] & \\
& = 2^{-n}\left[n(2^{n-1} -
  (\frac{1}{2}){n\choose \frac{n}{2}}) \right] +
2^{-n}\left[2n(2^{n-2} - {n-1\choose \frac{n}{2} - 1}) \right] & \\
 & = 2^{-n}\left[n(2^{n-1} - (\frac{1}{2}){n\choose \frac{n}{2}}) +
  2n(2^{n-2} - {n-1\choose \frac{n}{2} - 1}) \right] & \\
& = 2^{-n}{n\choose \frac{n}{2}}(\frac{n}{2}).
\end{eqnarray*}

\end{solution}

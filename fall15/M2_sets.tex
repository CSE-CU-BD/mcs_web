\begin{problem}

\bparts

\ppart
For each of the following sets, indicate whether it is
finite\inhandout{ (\textbf{F})},
countably infinite\inhandout{ (\textbf{C})},
or uncountable\inhandout{ (\textbf{U})}.

\begin{enumerate}

\item \label{itm:roots} The set of solutions to the equation $x^3 - x
  = -0.1$. \instatements{\hfill\examrule[0.4in]}

%\item \label{itm:naturals} The set of natural numbers $\naturals$. \instatements{\hfill\examrule[0.4in]}

%\item \label{itm:rationals} The set of rational numbers $\rationals$. \instatements{\hfill\examrule[0.4in]}

%\item \label{itm:reals} The set of real numbers $\reals$. \instatements{\hfill\examrule[0.4in]}

\item \label{itm:ints} The set of integers $\integers$. \instatements{\hfill\examrule[0.4in]}

\item \label{itm:complex} The set of complex numbers $\complexes$. \instatements{\hfill\examrule[0.4in]}

%\item \label{itm:char} The set of words in the English language no more
%  than 20 characters long. \instatements{\hfill\examrule[0.4in]}

\item \label{itm:self_bij} The powerset of the set of all possible
  bijections from $\set{1,2,\dots,10}$ to itself.
  \instatements{\hfill\examrule[0.4in]}

\item \label{itm:total_surj} An infinite set $S$ with the property
  that there exists a total surjective function $f : \naturals \to
  S$. \instatements{\hfill\examrule[0.4in]}

%\item \label{itm:union_count_uncount} A set $A \union B$ where $A$ is
%  countable and $B$ is uncountable.  \instatements{\hfill\examrule[0.4in]}

\end{enumerate}

\examspace[0.1in]

\ppart
Find an example of sets $A$ and $B$, such that $\naturals \strict A \strict B$.
\inhandout{($X \strict Y$ means there is no surjective
function from $X$ to $Y$.)}

\begin{solution}
Let $A$ be any uncountable set and $B$ be the power set of $A$, for
example, $A$ is the real numbers and $B \eqdef \power(\reals)$.
\end{solution}
